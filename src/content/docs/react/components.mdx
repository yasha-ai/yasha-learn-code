Компонент = [функция](/javascript/functions/), возвращающая [JSX](/react/jsx/).



## Интерактивный пример

<Playground client:visible
  template="react"
  files={{
    "/App.js": `import { useState } from 'react';

function Badge({ label, color }) {
  return (
    <span style={{ background: color || '#89b4fa', color: '#1e1e2e', borderRadius: 12, padding: '2px 10px', fontSize: 13, fontWeight: 'bold', margin: '0 3px' }}>
      {label}
    </span>
  );
}

function Card({ title, description, tags }) {
  return (
    <div style={{ background: '#313244', borderRadius: 8, padding: 16, margin: '10px 0' }}>
      <h3 style={{ margin: '0 0 8px', color: '#cdd6f4' }}>{title}</h3>
      <p style={{ margin: '0 0 10px', color: '#bac2de', fontSize: 14 }}>{description}</p>
      <div>{(tags || []).map(t => <Badge key={t} label={t} />)}</div>
    </div>
  );
}

function CardList({ cards }) {
  return <div>{cards.map((c, i) => <Card key={i} {...c} />)}</div>;
}

export default function App() {
  const [cards] = useState([
    { title: 'React', description: 'Библиотека для построения UI', tags: ['JS', 'Frontend'] },
    { title: 'Node.js', description: 'JavaScript на сервере', tags: ['JS', 'Backend'] },
    { title: 'TypeScript', description: 'Типизированный JavaScript', tags: ['Types', 'JS'] },
  ]);
  return (
    <div style={{ padding: 20, background: '#1e1e2e', color: '#cdd6f4', minHeight: '100vh', fontFamily: 'sans-serif' }}>
      <h2 style={{ margin: '0 0 4px' }}>Композиция компонентов</h2>
      <p style={{ color: '#bac2de', fontSize: 13, margin: '0 0 16px' }}>App → CardList → Card → Badge</p>
      <CardList cards={cards} />
    </div>
  );
}`,
  }}
/>
