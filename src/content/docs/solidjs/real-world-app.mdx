## Стор канбан-доски

```tsx
// stores/board.ts
import { createStore, produce } from 'solid-js/store';
import type { BoardStore, Card, Column } from '../types';

// Персистентный стор с localStorage
function createBoardStore() {
  const saved = localStorage.getItem('kanban-board');
  const initial: BoardStore = saved
    ? JSON.parse(saved)
    : {
        columns: [
          { id: 'todo', title: 'К выполнению', color: '#2c67d5', cardIds: [] },
          { id: 'progress', title: 'В работе', color: '#fb923c', cardIds: [] },
          { id: 'done', title: 'Готово', color: '#34d399', cardIds: [] },
        ],
        cards: {},
      };

  const [board, setBoard] = createStore<BoardStore>(initial);

  // Авто-сохранение при изменениях
  createEffect(() => {
    localStorage.setItem('kanban-board', JSON.stringify(board));
  });

  const addCard = (columnId: string, data: Omit<Card, 'id' | 'createdAt'>) => {
    const id = crypto.randomUUID();
    setBoard(produce(b => {
      b.cards[id] = { ...data, id, createdAt: Date.now() };
      b.columns.find(c => c.id === columnId)?.cardIds.push(id);
    }));
  };

  const moveCard = (cardId: string, fromColumnId: string, toColumnId: string) => {
    setBoard(produce(b => {
      const from = b.columns.find(c => c.id === fromColumnId);
      const to = b.columns.find(c => c.id === toColumnId);
      if (!from || !to) return;
      from.cardIds = from.cardIds.filter(id => id !== cardId);
      if (!to.cardIds.includes(cardId)) to.cardIds.push(cardId);
    }));
  };

  const deleteCard = (cardId: string) => {
    setBoard(produce(b => {
      b.columns.forEach(col => {
        col.cardIds = col.cardIds.filter(id => id !== cardId);
      });
      delete b.cards[cardId];
    }));
  };

  return { board, addCard, moveCard, deleteCard };
}

export const boardStore = createBoardStore();
```

## Компонент колонки

```tsx
// components/Column.tsx
function KanbanColumn(props: { column: Column; cards: Card[] }) {
  const [isDragOver, setIsDragOver] = createSignal(false);

  const handleDrop = (e: DragEvent) => {
    e.preventDefault();
    const cardId = e.dataTransfer?.getData('cardId');
    if (cardId) {
      boardStore.moveCard(cardId, /* fromColumnId */, props.column.id);
    }
    setIsDragOver(false);
  };

  return (
    <div
      class="column"
      classList={{ 'drag-over': isDragOver() }}
      onDragOver={e => { e.preventDefault(); setIsDragOver(true); }}
      onDragLeave={() => setIsDragOver(false)}
      onDrop={handleDrop}
    >
      <div class="column-header" style={{ 'border-color': props.column.color }}>
        <h3>{props.column.title}</h3>
        <span class="badge">{props.cards.length}</span>
      </div>

      <div class="cards-list">
        <For each={props.cards}>
          {card => (
            <KanbanCard
              card={card}
              columnId={props.column.id}
              onEdit={() => openEditModal(card)}
              onDelete={() => boardStore.deleteCard(card.id)}
              onMove={toColumnId => boardStore.moveCard(card.id, props.column.id, toColumnId)}
            />
          )}
        </For>
      </div>

      <button
        class="add-card-btn"
        onClick={() => openAddModal(props.column.id)}
      >
        + Добавить карточку
      </button>
    </div>
  );
}
```

```tsx
// Контекст позволяет любому компоненту в дереве получить доступ к store
const BoardContext = createContext<ReturnType<typeof createBoardStore>>();

export const useBoardStore = () => {
  const ctx = useContext(BoardContext);
  if (!ctx) throw new Error('useBoardStore must be used within BoardProvider');
  return ctx;
};

function BoardProvider(props: { children: JSXElement }) {
  const store = createBoardStore();
  return (
    <BoardContext.Provider value={store}>
      {props.children}
    </BoardContext.Provider>
  );
}

// Использование в App
function App() {
  return (
    <BoardProvider>
```

        <span style={{ fontSize: '10px', color: MUTED }}>{age === 0 ? 'сегодня' : age + ' дн. назад'}</span>
      </div>
    </div>
  );
}

// ──────────────────────────────────────────────────────────────────────────────
// COLUMN COMPONENT
// ──────────────────────────────────────────────────────────────────────────────
function Column({
  column, cards, columns, onAddCard, onEditCard, onDeleteCard, onMoveCard,
}: {
  column: KanbanColumn;
  cards: KanbanCard[];
  columns: KanbanColumn[];
  onAddCard: () => void;
  onEditCard: (card: KanbanCard) => void;
  onDeleteCard: (id: string) => void;
  onMoveCard: (cardId: string, toCol: string) => void;
}) {
  const [dragOver, setDragOver] = useState(false);

  return (
    <div
      style={{ minWidth: 240, width: 240, flexShrink: 0, display: 'flex', flexDirection: 'column', gap: '8px', background: CARD_BG, borderRadius: '12px', padding: '12px', border: '1px solid ' + (dragOver ? column.color + '66' : BORDER), transition: 'border-color .2s', maxHeight: 'calc(100vh - 140px)' }}
      onDragOver={e => { e.preventDefault(); setDragOver(true); }}
      onDragLeave={e => { if (!e.currentTarget.contains(e.relatedTarget as Node)) setDragOver(false); }}
      onDrop={e => {
        e.preventDefault();
        const cardId = e.dataTransfer?.getData('cardId');
        if (cardId) onMoveCard(cardId, column.id);
        setDragOver(false);
      }}
    >
      {/* Column header */}
      <div style={{ display: 'flex', alignItems: 'center', gap: '8px', paddingBottom: '8px', borderBottom: '1px solid ' + BORDER }}>
        <span>{column.icon}</span>
        <span style={{ fontSize: '13px', fontWeight: 600, color: TEXT, flex: 1 }}>{column.title}</span>
        <span style={{ fontSize: '11px', fontWeight: 700, color: 'white', background: column.color, borderRadius: '10px', padding: '1px 8px', minWidth: 20, textAlign: 'center' }}>
          {cards.length}
        </span>
      </div>

      {/* Cards */}
      <div style={{ display: 'flex', flexDirection: 'column', gap: '7px', overflowY: 'auto', flex: 1 }}>
        {cards.map(card => (
          <Card key={card.id} card={card} column={column} columns={columns}
            onEdit={() => onEditCard(card)}
            onDelete={() => onDeleteCard(card.id)}
            onMove={toCol => onMoveCard(card.id, toCol)}
          />
        ))}
        {cards.length === 0 && (
          <div style={{ padding: '16px', textAlign: 'center', color: MUTED, fontSize: '12px', border: '1px dashed ' + BORDER, borderRadius: '8px' }}>
            {dragOver ? '📥 Отпусти здесь' : 'Пусто'}
          </div>
        )}
      </div>

      {/* Add button */}
      <button onClick={onAddCard}
        style={{ padding: '8px', background: 'transparent', border: '1px dashed ' + BORDER, borderRadius: '7px', color: MUTED, cursor: 'pointer', fontSize: '12px', display: 'flex', alignItems: 'center', justifyContent: 'center', gap: '4px', transition: 'all .2s' }}
        onMouseEnter={e => { (e.currentTarget as HTMLButtonElement).style.borderColor = column.color; (e.currentTarget as HTMLButtonElement).style.color = column.color; }}
        onMouseLeave={e => { (e.currentTarget as HTMLButtonElement).style.borderColor = BORDER; (e.currentTarget as HTMLButtonElement).style.color = MUTED; }}
      >
        + Добавить
      </button>
    </div>
  );
}

// ──────────────────────────────────────────────────────────────────────────────
// STATS BAR
// ──────────────────────────────────────────────────────────────────────────────
function StatsBar({ board }: { board: BoardState }) {
  const total = Object.keys(board.cards).length;
  const highPriority = Object.values(board.cards).filter(c => c.priority === 'high').length;
  const done = board.columns.find(c => c.id === 'done')?.cardIds.length || 0;
  const pct = total > 0 ? Math.round((done / total) * 100) : 0;
  return (
    <div style={{ display: 'flex', gap: '16px', alignItems: 'center', flexWrap: 'wrap' }}>
      {[
        { label: 'Всего задач', value: total, color: TEXT },
        { label: 'Высокий приоритет', value: highPriority, color: RED },
        { label: 'Завершено', value: done + '/' + total, color: GREEN },
      ].map(s => (
        <div key={s.label} style={{ display: 'flex', alignItems: 'center', gap: '6px' }}>
          <span style={{ fontSize: '16px', fontWeight: 700, color: s.color }}>{s.value}</span>
          <span style={{ fontSize: '11px', color: MUTED }}>{s.label}</span>
        </div>
      ))}
      <div style={{ flex: 1, minWidth: 80 }}>
        <div style={{ height: '4px', background: BORDER, borderRadius: '2px', overflow: 'hidden' }}>
          <div style={{ height: '100%', background: GREEN, width: pct + '%', transition: 'width .3s', borderRadius: '2px' }} />
        </div>
        <div style={{ fontSize: '10px', color: MUTED, marginTop: '2px', textAlign: 'right' }}>{pct}% готово</div>
      </div>
    </div>
  );
}

    );

          Сбросить
        </button>
      </div>

      {/* Stats */}

        <StatsBar board={board} />
      </div>

      {/* Board */}
      <div style={{ flex: 1, overflowX: 'auto', padding: '16px' }}>
        <div style={{ display: 'flex', gap: '12px', minWidth: 'max-content' }}>

                onEditCard={openEdit}
                onDeleteCard={deleteCard}
                onMoveCard={moveCard}

            );
          })}
        </div>

          <div style={{ textAlign: 'center', padding: '40px', color: MUTED, fontSize: '14px' }}>
            Ничего не найдено по запросу «{search}»
          </div>
        )}
      </div>

      {/* Modal */}
      {modal && (
        <CardModal
          state={modal}
          columns={board.columns}
          onClose={() => setModal(null)}
          onAdd={addCard}
          onUpdate={updateCard}
        />
      )}
    </div>
  );
}

  }}
