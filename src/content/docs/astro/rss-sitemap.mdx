Хорошо оптимизированный сайт должен быть видим как для пользователей, так и для поисковых роботов и RSS-ридеров. Astro предоставляет официальные интеграции для генерации RSS-лент и карты сайта, а также набор паттернов для SEO.

## RSS-лента с `@astrojs/rss`

RSS позволяет подписчикам получать обновления вашего блога без посещения сайта. Устанавливаем интеграцию:

```bash
npm install @astrojs/rss
```

Создаём файл-эндпоинт `src/pages/rss.xml.ts`:

```typescript
// src/pages/rss.xml.ts
import rss from '@astrojs/rss';
import { getCollection } from 'astro:content';
import type { APIContext } from 'astro';

export async function GET(context: APIContext) {
  const posts = await getCollection('blog', ({ data }) => data.published);
  
  return rss({
    title: 'Мой блог об Astro',
    description: 'Свежие статьи о веб-разработке и Astro',
    site: context.site!,
    items: posts.map(post => ({
      title: post.data.title,
      description: post.data.description,
      pubDate: post.data.publishedAt,
      link: `/blog/${post.slug}/`,
      // Опционально: полный HTML контент
      content: post.body,
    })),
    // Кастомные XML-элементы
    customData: `<language>ru-ru</language>`,
  });
}
```

Лента будет доступна по адресу `/rss.xml`. Добавьте ссылку в `<head>`:

```astro
<link rel="alternate" type="application/rss+xml" title="RSS" href="/rss.xml" />
```

## Sitemap с `@astrojs/sitemap`

Карта сайта помогает поисковым роботам эффективно индексировать все ваши страницы:

```bash
npx astro add sitemap
```

Регистрируем интеграцию в `astro.config.mjs`:

```javascript
import { defineConfig } from 'astro/config';
import sitemap from '@astrojs/sitemap';

export default defineConfig({
  site: 'https://mysite.com',
  integrations: [
    sitemap({
      changefreq: 'weekly',
      priority: 0.7,
      lastmod: new Date(),
      // Исключаем служебные страницы
      filter: (page) => !page.includes('/admin/'),
      // Кастомные настройки для конкретных страниц
      customPages: ['https://mysite.com/custom-page'],
      serialize(item) {
        // Повышаем приоритет главной страницы
        if (item.url === 'https://mysite.com/') {
          return { ...item, priority: 1.0 };
        }
        return item;
      },
    }),
  ],
});
```

После билда файлы `sitemap-index.xml` и `sitemap-0.xml` появятся в папке `dist/`.

## Файл `robots.txt`

Создаём статический файл `public/robots.txt`:

```
User-agent: *
Allow: /
Disallow: /admin/
Disallow: /api/

Sitemap: https://mysite.com/sitemap-index.xml
```

## Open Graph для социальных сетей

Теги Open Graph контролируют, как ваш контент выглядит при публикации в VK, Telegram, Twitter и других платформах:

```astro
---
const { title, description, image, url } = Astro.props;
const ogImage = new URL(image, Astro.site);
---
<head>
  <!-- Основные OG теги -->
  <meta property="og:type" content="article" />
  <meta property="og:title" content={title} />
  <meta property="og:description" content={description} />
  <meta property="og:image" content={ogImage} />
  <meta property="og:url" content={url} />

  <!-- Twitter Cards -->
  <meta name="twitter:card" content="summary_large_image" />
  <meta name="twitter:title" content={title} />
  <meta name="twitter:description" content={description} />
  <meta name="twitter:image" content={ogImage} />
</head>
```

function generateRSS(posts: Post[]): string {

function generateSitemap(posts: Post[]): string {
  const staticPages = ['', '/about', '/blog', '/contact'];
  const allUrls = [
    ...staticPages.map((path, i) => ({ url: SITE + path + '/', priority: i === 0 ? '1.0' : '0.7', freq: 'weekly' })),
    ...posts.map(p => ({ url: \`\${SITE}/blog/\${p.slug}/\`, priority: '0.8', freq: 'monthly', lastmod: p.date })),
  ];

  const urlEntries = allUrls.map(u => \`  <url>
    <loc>\${u.url}</loc>
    <changefreq>\${u.freq}</changefreq>
    <priority>\${u.priority}</priority>\${u.lastmod ? '\\n    <lastmod>' + u.lastmod + '</lastmod>' : ''}
  </url>\`).join('\\n');

  return \`<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
\${urlEntries}
</urlset>\`;
}

function XmlHighlight({ code }: { code: string }) {
  const lines = code.split('\\n');
  return (
    <pre style={{
      margin: 0,
      fontFamily: 'monospace',
      fontSize: 12,
      lineHeight: 1.65,
      overflowX: 'auto',
      padding: 16,
    }}>
      {lines.map((line, i) => {
        const colored = line
          .replace(/(&lt;[!?][^>]*&gt;)/g, '<span style="color:#6366f1">$1</span>')
          .replace(/(&lt;\\/[a-zA-Z:]+&gt;)/g, '<span style="color:#7dd3fc">$1</span>')
          .replace(/(&lt;[a-zA-Z:]+)/g, '<span style="color:#7dd3fc">$1</span>')
          .replace(/(&gt;)/g, '<span style="color:#7dd3fc">$1</span>')
          .replace(/([a-zA-Z:]+)(=)/g, '<span style="color:#86efac">$1</span><span style="color:#e2e8f0">$2</span>')
          .replace(/("([^"]*)")/g, '<span style="color:#fde047">"$2"</span>')
          .replace(/(<!\[CDATA\[)(.*?)(\]\]>)/g, '$1<span style="color:#e2e8f0">$2</span>$3');
        return (
          <div key={i} style={{ display: 'flex', gap: 12 }}>
            <span style={{ color: '#334155', userSelect: 'none', minWidth: 28, textAlign: 'right' }}>{i + 1}</span>
            <span style={{ color: TEXT }} dangerouslySetInnerHTML={{ __html: line
              .replace(/</g, '&lt;')
              .replace(/>/g, '&gt;')
            }} />
          </div>
        );
      })}
    </pre>
  );
}
