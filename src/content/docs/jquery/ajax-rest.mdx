<Playground client:visible
  template="static"
  files={{
    "/index.html": `<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
  <style>
    body { background: #1e1e2e; color: #cdd6f4; font-family: sans-serif; padding: 20px; margin: 0; }
    button { background: #89b4fa; color: #1e1e2e; border: none; padding: 6px 12px; border-radius: 6px; cursor: pointer; margin: 2px; font-size: 13px; }
    button:hover { background: #74c7ec; }
    button.delete { background: #f38ba8; }
    button.edit { background: #fab387; }
    input { background: #313244; color: #cdd6f4; border: 1px solid #45475a; padding: 6px 10px; border-radius: 4px; width: 180px; }
    .box { background: #313244; border-radius: 8px; padding: 16px; margin: 10px 0; }
    .item { background: #45475a; border-radius: 6px; padding: 8px 12px; margin: 4px 0; display: flex; align-items: center; justify-content: space-between; }
    .badge { font-size: 11px; background: #313244; padding: 2px 7px; border-radius: 10px; color: #89b4fa; }
    #log { background: #181825; border-radius: 6px; padding: 10px; font-family: monospace; font-size: 12px; color: #a6e3a1; min-height: 30px; }
  </style>
</head>
<body>
  <h3 style="color:#cba6f7">REST API — симуляция CRUD</h3>
  <div class="box">
    <input id="newName" type="text" placeholder="Имя нового ресурса">
    <button id="btnCreate">POST (Create)</button>
    <button id="btnGetAll">GET (Read All)</button>
  </div>
  <div id="list" class="box" style="min-height:60px"></div>
  <div class="box">
    <strong style="font-size:12px;color:#bac2de">Лог запросов:</strong>
    <div id="log">—</div>
  </div>
  <script>
    var db = [
      { id: 1, name: 'Ресурс Alpha' },
      { id: 2, name: 'Ресурс Beta' }
    ];
    var nextId = 3;

    function logReq(method, url, color) {
      color = color || '#a6e3a1';
      $('#log').html('<span style="color:' + color + '">' + method + '</span> ' + url + ' → ' + (color === '#f38ba8' ? '404' : '200 OK'));
    }

    function renderList() {
      if (!db.length) { $('#list').html('<span style="color:#585b70">Список пуст</span>'); return; }
      var html = '';
      $.each(db, function(i, item) {
        html += '<div class="item" data-id="' + item.id + '">' +
          '<span><span class="badge">ID:' + item.id + '</span> ' + item.name + '</span>' +
          '<span>' +
          '<button class="edit" data-id="' + item.id + '">PUT</button> ' +
          '<button class="delete" data-id="' + item.id + '">DELETE</button>' +
          '</span></div>';
      });
      $('#list').html(html);
    }

    function delay(fn) { setTimeout(fn, 400); }

    $('#btnCreate').click(function() {
      var name = $('#newName').val().trim() || 'Новый ресурс';
      delay(function() {
        var item = { id: nextId++, name: name };
        db.push(item);
        logReq('POST', '/api/resources');
        renderList();
        $('#newName').val('');
      });
    });

    $('#btnGetAll').click(function() {
      delay(function() {
        logReq('GET', '/api/resources');
        renderList();
      });
    });

    $(document).on('click', '.edit', function() {
      var id = parseInt($(this).data('id'));
      var item = $.grep(db, function(x) { return x.id === id; })[0];
      if (!item) return;
      var newName = prompt('Новое имя для "' + item.name + '":', item.name);
      if (newName) {
        delay(function() {
          item.name = newName;
          logReq('PUT', '/api/resources/' + id);
          renderList();
        });
      }
    });

    $(document).on('click', '.delete', function() {
      var id = parseInt($(this).data('id'));
      delay(function() {
        db = $.grep(db, function(x) { return x.id !== id; });
        logReq('DELETE', '/api/resources/' + id, '#f38ba8');
        renderList();
      });
    });

    renderList();
  </script>
</body>
</html>`,
  }}
/>
