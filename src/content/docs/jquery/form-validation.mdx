Проверка пользовательского ввода перед отправкой.

## Ручная валидация

Свои проверки

```javascript
// Пример кода для Ручная валидация
$('.example').text('Hello from Ручная валидация');
```

## jQuery Validation Plugin

Готовое решение

```javascript
// Пример кода для jQuery Validation Plugin
$('.example').text('Hello from jQuery Validation Plugin');
```

## Кастомные правила

Добавление своих проверок

```javascript
// Пример кода для Кастомные правила
$('.example').text('Hello from Кастомные правила');
```

## UX валидации

Когда показывать ошибки

```javascript
// Пример кода для UX валидации
$('.example').text('Hello from UX валидации');
```

  template="static"
  files={{
    "/index.html": `<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
  <style>
    body { background: #1e1e2e; color: #cdd6f4; font-family: sans-serif; padding: 20px; margin: 0; }
    button { background: #89b4fa; color: #1e1e2e; border: none; padding: 10px 20px; border-radius: 6px; cursor: pointer; font-weight: bold; }
    .box { background: #313244; border-radius: 8px; padding: 20px; }
    input { background: #181825; color: #cdd6f4; border: 1px solid #45475a; padding: 8px 10px; border-radius: 4px; width: 100%; box-sizing: border-box; margin-bottom: 4px; }
    input.error { border-color: #f38ba8; }
    input.ok { border-color: #a6e3a1; }
    .err-msg { color: #f38ba8; font-size: 12px; min-height: 16px; margin-bottom: 10px; }
    label { font-size: 13px; color: #bac2de; display: block; margin-bottom: 4px; }
    .success { color: #a6e3a1; font-weight: bold; text-align: center; padding: 20px; font-size: 18px; }
  </style>
</head>
<body>
  <div class="box">
    <h3 style="margin:0 0 16px">Live Form Validation</h3>
    <form id="myForm">
      <label>Имя (мин. 2 символа)</label>
      <input id="name" type="text" placeholder="Введите имя">
      <div class="err-msg" id="nameErr"></div>
      <label>Email</label>
      <input id="email" type="text" placeholder="example@mail.com">
      <div class="err-msg" id="emailErr"></div>
      <label>Пароль (мин. 6 символов)</label>
      <input id="password" type="password" placeholder="Пароль">
      <div class="err-msg" id="passErr"></div>
      <button type="submit">Отправить</button>
    </form>
    <div class="success" id="successMsg" style="display:none">✅ Форма отправлена!</div>
  </div>
  <script>
    $(function() {
      function validateName() {
        var v = $('#name').val().trim();
        if (!v) { $('#nameErr').text('Имя обязательно'); $('#name').removeClass('ok').addClass('error'); return false; }
        if (v.length < 2) { $('#nameErr').text('Мин. 2 символа'); $('#name').removeClass('ok').addClass('error'); return false; }
        $('#nameErr').text(''); $('#name').removeClass('error').addClass('ok'); return true;
      }
      function validateEmail() {
        var v = $('#email').val().trim();
        if (!v) { $('#emailErr').text('Email обязателен'); $('#email').removeClass('ok').addClass('error'); return false; }
        if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v)) { $('#emailErr').text('Неверный формат'); $('#email').removeClass('ok').addClass('error'); return false; }
        $('#emailErr').text(''); $('#email').removeClass('error').addClass('ok'); return true;
      }
      function validatePass() {
        var v = $('#password').val();
        if (v.length < 6) { $('#passErr').text('Мин. 6 символов'); $('#password').removeClass('ok').addClass('error'); return false; }
        $('#passErr').text(''); $('#password').removeClass('error').addClass('ok'); return true;
      }
      $('#name').on('input', validateName);
      $('#email').on('input', validateEmail);
      $('#password').on('input', validatePass);
      $('#myForm').on('submit', function(e) {
        e.preventDefault();
        var ok = validateName() & validateEmail() & validatePass();
        if (ok) { $('#myForm').hide(); $('#successMsg').show(); }
      });
    });
  </script>
</body>
</html>`,
  }}
/>
