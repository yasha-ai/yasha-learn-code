jQuery упрощает изменение стилей элементов.

## Метод .css()

Чтение и установка стилей

```javascript
// Пример кода для Метод .css()
$('.example').text('Hello from Метод .css()');
```

## Классы

.addClass(), .removeClass(), .toggleClass(), .hasClass()

```javascript
// Пример кода для Классы
$('.example').text('Hello from Классы');
```

## Размеры

.width(), .height(), .innerWidth(), .outerWidth()

```javascript
// Пример кода для Размеры
$('.example').text('Hello from Размеры');
```

## Позиция

.offset(), .position(), .scrollTop(), .scrollLeft()

```javascript
// Пример кода для Позиция
$('.example').text('Hello from Позиция');
```

  template="static"
  files={{
    "/index.html": `<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
  <style>
    body { background: #1e1e2e; color: #cdd6f4; font-family: sans-serif; padding: 20px; margin: 0; }
    button { background: #89b4fa; color: #1e1e2e; border: none; padding: 7px 13px; border-radius: 6px; cursor: pointer; margin: 3px; font-size: 13px; }
    button:hover { background: #74c7ec; }
    .box { background: #313244; border-radius: 8px; padding: 16px; margin: 10px 0; }
    #demo { background: #45475a; border-radius: 8px; padding: 24px; text-align: center; font-size: 18px; transition: all 0.3s; border: 2px solid transparent; }
    .bold-text { font-weight: bold; }
    .big-text { font-size: 24px; }
    .rounded { border-radius: 20px; }
    .bordered { border: 2px solid #89b4fa; }
    .shadow { box-shadow: 0 4px 20px rgba(137,180,250,0.4); }
    .inverted { background: #89b4fa !important; color: #1e1e2e !important; }
    .output { background: #181825; border-radius: 6px; padding: 10px; font-family: monospace; font-size: 12px; color: #a6e3a1; margin-top: 8px; min-height: 28px; }
    input[type=color] { border: none; background: none; cursor: pointer; width: 36px; height: 28px; padding: 0; vertical-align: middle; }
    input[type=range] { vertical-align: middle; width: 120px; }
  </style>
</head>
<body>
  <h3 style="color:#cba6f7">Работа с CSS</h3>
  <div class="box">
    <div id="demo">Демо-элемент</div>
  </div>
  <div class="box">
    <div>
      <strong style="font-size:12px;color:#bac2de">Классы:</strong><br>
      <button id="addBold">.addClass('bold-text')</button>
      <button id="removeBold">.removeClass('bold-text')</button>
      <button id="toggleBig">.toggleClass('big-text')</button>
      <button id="toggleRounded">.toggleClass('rounded')</button>
      <button id="toggleBordered">.toggleClass('bordered')</button>
      <button id="toggleShadow">.toggleClass('shadow')</button>
      <button id="toggleInverted">.toggleClass('inverted')</button>
    </div>
    <div style="margin-top:10px">
      <strong style="font-size:12px;color:#bac2de">.css() — прямые стили:</strong><br>
      Цвет фона: <input type="color" id="bgColor" value="#45475a">
      Прозрачность: <input type="range" id="opacity" min="10" max="100" value="100">
      <button id="resetCss">Сброс</button>
    </div>
    <div class="output" id="out">Нажмите кнопку...</div>
  </div>
  <script>
    var $d = $('#demo');
    function log(msg) { $('#out').text(msg); }

    $('#addBold').click(function()       { $d.addClass('bold-text');       log(".addClass('bold-text') — hasClass: " + $d.hasClass('bold-text')); });
    $('#removeBold').click(function()    { $d.removeClass('bold-text');    log(".removeClass('bold-text') — hasClass: " + $d.hasClass('bold-text')); });
    $('#toggleBig').click(function()     { $d.toggleClass('big-text');     log(".toggleClass('big-text') — сейчас: " + $d.hasClass('big-text')); });
    $('#toggleRounded').click(function() { $d.toggleClass('rounded');      log(".toggleClass('rounded')"); });
    $('#toggleBordered').click(function(){ $d.toggleClass('bordered');     log(".toggleClass('bordered')"); });
    $('#toggleShadow').click(function()  { $d.toggleClass('shadow');       log(".toggleClass('shadow')"); });
    $('#toggleInverted').click(function(){ $d.toggleClass('inverted');     log(".toggleClass('inverted')"); });

    $('#bgColor').on('input', function() {
      $d.css('background-color', this.value);
      log(".css('background-color', '" + this.value + "')");
    });
    $('#opacity').on('input', function() {
      $d.css('opacity', this.value / 100);
      log(".css('opacity', " + (this.value / 100) + ")");
    });
    $('#resetCss').click(function() {
      $d.removeAttr('style').removeClass('bold-text big-text rounded bordered shadow inverted');
      log('Стили и классы сброшены');
    });
  </script>
</body>
</html>`,
  }}
/>
