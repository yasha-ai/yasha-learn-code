  files={{

<html lang="en">
<head>

    button { background: #89b4fa; color: #1e1e2e; border: none; padding: 8px 16px; border-radius: 6px; cursor: pointer; margin: 4px; }
    button:hover { background: #74c7ec; }
    .box { background: #313244; border-radius: 8px; padding: 16px; margin: 10px 0; }
    .output { background: #181825; border-radius: 6px; padding: 12px; font-family: monospace; font-size: 13px; color: #a6e3a1; white-space: pre-wrap; min-height: 40px; }
    span.item-tag { display: inline-block; background: #45475a; border-radius: 4px; padding: 3px 7px; margin: 2px; font-size: 11px; }
  </style>
</head>
<body>
  <div id="hiddenItems" style="display:none"></div>
  <div class="box">
    <h3 style="margin:0 0 8px">Кэширование селекторов</h3>
    <button id="btnUncached">❌ Без кэша (500 поисков)</button>
    <button id="btnCached">✅ С кэшем (500 обращений)</button>
    <div class="output" id="result">Нажми кнопку для сравнения...</div>
  </div>
  <div class="box">
    <h3 style="margin:0 0 8px">Batch DOM обновление (одна операция)</h3>
    <button id="btnBatch">Добавить 50 элементов</button>
    <div id="container" style="margin-top:8px"></div>
  </div>
  <script>
    $(function() {
      for (var i = 0; i < 200; i++) {
        $('#hiddenItems').append('<div class="perf-item">item</div>');
      }
      $('#btnUncached').click(function() {
        var t = performance.now();
        for (var i = 0; i < 500; i++) { $('.perf-item').length; }
        var ms = (performance.now() - t).toFixed(2);
        $('#result').text('Без кэша: DOM обходится 500 раз\nВремя: ' + ms + ' мс\n\nКод:\nfor(i=0;i<500;i++) { $(".perf-item").length; }');
      });
      $('#btnCached').click(function() {
        var t = performance.now();
        var $items = $('.perf-item');
        for (var i = 0; i < 500; i++) { $items.length; }
        var ms = (performance.now() - t).toFixed(2);
        $('#result').text('С кэшем: DOM обходится 1 раз\nВремя: ' + ms + ' мс\n\nКод:\nvar $items = $(".perf-item"); // кэш!\nfor(i=0;i<500;i++) { $items.length; }');
      });
      $('#btnBatch').click(function() {
        $('#container').empty();
        var html = '';
        for (var i = 1; i <= 50; i++) html += '<span class="item-tag">' + i + '</span>';
        $('#container').html(html);
        $(this).text('✅ 50 элементов за 1 DOM операцию');
      });
    });
  </script>
</body>
</html>`,
  }}
/>
