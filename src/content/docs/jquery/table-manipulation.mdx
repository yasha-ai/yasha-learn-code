  </style>
</head>
<body>
  <div class="box">
    <input id="filterInput" placeholder="🔍 Фильтр по имени...">
    <button id="addRow">+ Добавить строку</button>
    <table id="myTable">
      <thead><tr>
        <th data-col="0">Имя ↕</th>
        <th data-col="1">Возраст ↕</th>
        <th data-col="2">Город ↕</th>
      </tr></thead>
      <tbody>
        <tr><td>Алекс</td><td>28</td><td>Москва</td></tr>
        <tr><td>Мария</td><td>34</td><td>СПб</td></tr>
        <tr><td>Дмитрий</td><td>22</td><td>Казань</td></tr>
        <tr><td>Елена</td><td>45</td><td>Москва</td></tr>
        <tr><td>Игорь</td><td>31</td><td>Новосибирск</td></tr>
      </tbody>
    </table>
  </div>
  <script>
    $(function() {
      var sortDir = {};
      var names = ['Сергей','Анна','Павел','Ольга','Тимур'];
      var cities = ['Москва','СПб','Сочи','Екб','Уфа'];
      $('#filterInput').on('input', function() {
        var q = $(this).val().toLowerCase();
        $('#myTable tbody tr').each(function() {
          $(this).toggle($(this).find('td:first').text().toLowerCase().indexOf(q) >= 0);
        });
      });
      $('#myTable th').click(function() {
        var col = +$(this).data('col');
        sortDir[col] = !sortDir[col];
        var rows = $('#myTable tbody tr').toArray();
        rows.sort(function(a, b) {
          var av = $(a).find('td').eq(col).text();
          var bv = $(b).find('td').eq(col).text();
          var n = parseFloat(av) - parseFloat(bv);
          var cmp = isNaN(n) ? av.localeCompare(bv) : n;
          return sortDir[col] ? cmp : -cmp;
        });
        $.each(rows, function(i, r) { $('#myTable tbody').append(r); });
      });
      $('#addRow').click(function() {
        var n = names[Math.floor(Math.random()*names.length)];
        var a = 20 + Math.floor(Math.random()*30);
        var c = cities[Math.floor(Math.random()*cities.length)];
        $('#myTable tbody').append('<tr><td>' + n + '</td><td>' + a + '</td><td>' + c + '</td></tr>');
      });
    });
  </script>
</body>
</html>`,
  }}
/>
