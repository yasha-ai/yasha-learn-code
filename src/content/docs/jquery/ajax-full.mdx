Самый мощный метод для AJAX запросов.

## Базовый синтаксис

url, method, data, success

```javascript
// Пример кода для Базовый синтаксис
$('.example').text('Hello from Базовый синтаксис');
```

## Параметры запроса

headers, dataType, contentType

```javascript
// Пример кода для Параметры запроса
$('.example').text('Hello from Параметры запроса');
```

## Обработка ответа

success, error, complete

```javascript
// Пример кода для Обработка ответа
$('.example').text('Hello from Обработка ответа');
```

## Promise интерфейс

.done(), .fail(), .always()

```javascript
// Пример кода для Promise интерфейс
$('.example').text('Hello from Promise интерфейс');
```

    <select id="dtype">
      <option value="json">json</option>

    </select>
    <label>timeout (ms): </label>
    <input id="timeout" type="number" value="2000" style="width:70px">

    $('#runAjax').click(function() {
      var method = $('#method').val();
      var dtype = $('#dtype').val();
      var timeout = parseInt($('#timeout').val()) || 2000;
      var payload = $('#payload').val();
      var $log = $('#log');

      $log.text('beforeSend: запрос подготовлен\\nmethod: ' + method + ', dataType: ' + dtype + ', timeout: ' + timeout + 'ms');

      // Simulate $.ajax options display + mock response
      var deferred = $.Deferred();
      var mockData = { status: 'ok', method: method, received: payload, timestamp: Date.now() };

      setTimeout(function() {
        if (timeout < 500) {
          deferred.reject('timeout');
        } else {
          deferred.resolve(mockData);
        }
      }, 600);

      $.ajax({

        method: method,
        dataType: dtype,
        timeout: timeout,
        data: payload,
        beforeSend: function() {
          $log.append('\\nbeforeSend: заголовки отправлены');
        }
      }).always(function() {});  // показать синтаксис

      deferred.promise()
        .done(function(data) {
          $log.append('\\nsuccess / .done():');
          $log.append('\\n' + JSON.stringify(data, null, 2));
          $log.append('\\ncomplete / .always(): запрос завершён');
        })
        .fail(function(reason) {
          $log.append('\\nerror / .fail(): ' + reason);
          $log.append('\\ncomplete / .always(): запрос завершён');
        });
    });
  </script>
