Эффективная обработка событий для динамических элементов.

## Что это

Event delegation концепция

```javascript
// Пример кода для Что это
$('.example').text('Hello from Что это');
```

## .on() с делегацией

Синтаксис

```javascript
// Пример кода для .on() с делегацией
$('.example').text('Hello from .on() с делегацией');
```

## Производительность

Почему это быстрее

```javascript
// Пример кода для Производительность
$('.example').text('Hello from Производительность');
```

## Практика

Динамические списки

```javascript
// Пример кода для Практика
$('.example').text('Hello from Практика');
```

  <div class="box">
    <p style="margin:0 0 10px;font-size:14px">Кликайте по элементам, затем добавляйте новые — делегирование работает для всех!</p>
    <button id="addItem">+ Добавить элемент</button>
    <div id="list" style="margin-top:10px">
      <div class="item">Элемент 1</div>
      <div class="item">Элемент 2</div>

    </div>

  </div>
  <script>
    $(function() {
      var count = 3;
      // Делегирование: обработчик на #list, срабатывает на .item
      $('#list').on('click', '.item', function() {
        $('#output').text('Клик по: "' + $(this).text() + '"');
        var $el = $(this);
        $el.css('background', '#89b4fa').css('color', '#1e1e2e');
        setTimeout(function() { $el.css('background', '').css('color', ''); }, 600);
      });
      $('#addItem').click(function() {
        count++;
        $('#list').append('<div class="item">Элемент ' + count + ' (добавлен динамически)</div>');
      });
    });
  </script>
</body>

  }}
