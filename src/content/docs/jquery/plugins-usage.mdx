jQuery имеет огромную экосистему готовых плагинов.

## Где искать плагины

jQuery Plugin Registry

```javascript
// Пример кода для Где искать плагины
$('.example').text('Hello from Где искать плагины');
```

## Подключение

Script tags, NPM

```javascript
// Пример кода для Подключение
$('.example').text('Hello from Подключение');
```

## Популярные плагины

Slider, Lightbox, Validation

```javascript
// Пример кода для Популярные плагины
$('.example').text('Hello from Популярные плагины');
```

## Настройка плагинов

Options и callbacks

```javascript
// Пример кода для Настройка плагинов
$('.example').text('Hello from Настройка плагинов');
```

  <div class="box">
    <h3 style="margin:0 0 8px">Плагин: $.fn.tagSelect</h3>
    <p style="font-size:13px;color:#bac2de;margin:0 0 10px">Кликай по тегам — плагин обрабатывает выбор</p>
    <div id="tagContainer">
      <span class="tag">JavaScript</span>
      <span class="tag">jQuery</span>
      <span class="tag">HTML</span>
      <span class="tag">CSS</span>
      <span class="tag">React</span>
      <span class="tag">Node.js</span>
    </div>
    <div style="margin-top:10px">
      <button id="getSelected">Показать выбранные</button>
      <button id="clearAll">Очистить</button>
    </div>
    <div class="output" id="result">Выбери теги и нажми кнопку...</div>
  </div>
  <script>
    // Плагин для выбора тегов
    $.fn.tagSelect = function(options) {
      var opts = $.extend({ onSelect: null }, options);
      return this.each(function() {
        $(this).on('click', '.tag', function() {
          $(this).toggleClass('selected');
          if (opts.onSelect) opts.onSelect($(this).text(), $(this).hasClass('selected'));
        });
      });
    };
    $(function() {
      $('#tagContainer').tagSelect({
        onSelect: function(tag, selected) {
          console.log(tag + (selected ? ' selected' : ' deselected'));
        }
      });
      $('#getSelected').click(function() {
        var selected = $('.tag.selected').map(function() { return $(this).text(); }).get();
        $('#result').text(selected.length ? 'Выбрано: ' + selected.join(', ') : 'Ничего не выбрано');
      });
      $('#clearAll').click(function() {
        $('.tag').removeClass('selected');
        $('#result').text('Очищено');
      });
    });
  </script>
</body>
</html>`,
  }}
