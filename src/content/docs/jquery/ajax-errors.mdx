<Playground client:visible
  template="static"
  files={{
    "/index.html": `<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
  <style>
    body { background: #1e1e2e; color: #cdd6f4; font-family: sans-serif; padding: 20px; margin: 0; }
    button { background: #89b4fa; color: #1e1e2e; border: none; padding: 8px 16px; border-radius: 6px; cursor: pointer; margin: 4px; font-size: 14px; }
    button:hover { background: #74c7ec; }
    .box { background: #313244; border-radius: 8px; padding: 16px; margin: 10px 0; }
    .output { background: #181825; border-radius: 6px; padding: 12px; font-family: monospace; font-size: 13px; min-height: 40px; color: #a6e3a1; }
    .error { color: #f38ba8; }
    .warn { color: #fab387; }
  </style>
</head>
<body>
  <h3 style="color:#cba6f7">Обработка ошибок AJAX</h3>
  <div class="box">
    <button id="sim404">Симулировать 404</button>
    <button id="sim500">Симулировать 500</button>
    <button id="simTimeout">Симулировать Timeout</button>
    <button id="simSuccess">Успешный запрос</button>
  </div>
  <div class="box">
    <div id="output" class="output">Нажмите кнопку для симуляции запроса...</div>
  </div>
  <script>
    function simulateAjax(scenario) {
      var $out = $('#output');
      $out.removeClass('error warn').text('Отправляю запрос...').css('color','#cdd6f4');

      // Simulated $.ajax using setTimeout + mock deferred
      var deferred = $.Deferred();

      setTimeout(function() {
        if (scenario === 'success') {
          deferred.resolve({message: 'Данные получены!'}, 'success', {status: 200});
        } else if (scenario === '404') {
          deferred.reject({status: 404, statusText: 'Not Found'}, 'error', 'Not Found');
        } else if (scenario === '500') {
          deferred.reject({status: 500, statusText: 'Internal Server Error'}, 'error', 'Internal Server Error');
        } else if (scenario === 'timeout') {
          deferred.reject({status: 0, statusText: 'timeout'}, 'timeout', 'timeout');
        }
      }, 800);

      deferred.promise()
        .done(function(data) {
          $out.css('color','#a6e3a1').text('✅ Успех (200): ' + JSON.stringify(data));
        })
        .fail(function(jqXHR, textStatus, errorThrown) {
          var msg = '';
          if (textStatus === 'timeout') {
            msg = '⏱ Timeout: запрос превысил время ожидания';
          } else if (jqXHR.status === 404) {
            msg = '❌ 404 Not Found: ресурс не существует';
          } else if (jqXHR.status === 500) {
            msg = '🔥 500 Server Error: внутренняя ошибка сервера';
          }
          $out.css('color','#f38ba8').text(msg);
        })
        .always(function() {
          $out.append('\\n— always() выполнен');
        });
    }

    $('#sim404').click(function() { simulateAjax('404'); });
    $('#sim500').click(function() { simulateAjax('500'); });
    $('#simTimeout').click(function() { simulateAjax('timeout'); });
    $('#simSuccess').click(function() { simulateAjax('success'); });
  </script>
</body>
</html>`,
  }}
/>
