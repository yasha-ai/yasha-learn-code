Две самые фундаментальные настройки Webpack — `entry` и `output`. Они определяют **откуда** Webpack начинает сборку и **куда** записывает результат. Понимание всех вариантов конфигурации этих полей открывает возможности для создания сложных multi-page или multi-bundle приложений.

## Entry — точка входа

### Строка (простейший вариант)
```js
module.exports = {
  entry: './src/index.js',
};
// Webpack создаст один чанк с именем 'main'
```

### Массив — несколько файлов в один бандл
```js
module.exports = {
  entry: ['./src/polyfills.js', './src/index.js'],
};
// Оба файла попадут в один бандл 'main'
// Используется для добавления polyfills
```

### Объект — множество точек входа
```js
module.exports = {
  entry: {
    main: './src/index.js',
    admin: './src/admin/index.js',
    vendor: ['react', 'react-dom'],
  },
};
// Создаст main.js, admin.js, vendor.js
```

### Динамический entry с функцией
```js
module.exports = {
  entry: () => new Promise((resolve) => {
    resolve({
      main: './src/index.js',
      admin: './src/admin.js',
    });
  }),
};
```

### Entry дескриптор (Webpack 5)
```js
module.exports = {
  entry: {
    main: {
      import: './src/index.js',
      dependOn: 'shared',         // зависит от shared чанка
      filename: 'pages/[name].js', // кастомный путь вывода
    },
    shared: ['react', 'react-dom'],
  },
};
```

## Output — настройки вывода

### Базовые настройки
```js
output: {
  path: path.resolve(__dirname, 'dist'), // абсолютный путь!
  filename: 'bundle.js',                  // для одного entry
  filename: '[name].bundle.js',           // для множества entry
  publicPath: '/',                        // базовый URL для ресурсов
  clean: true,                            // очищать dist/ (Webpack 5)
}
```

### Шаблонные строки (Template Strings)

| Паттерн | Описание | Пример |
|---------|----------|--------|
| `[name]` | Имя чанка/entry | `main` |
| `[id]` | Числовой ID чанка | `0` |
| `[hash]` | Хеш всей сборки | `abc123` |
| `[chunkhash]` | Хеш конкретного чанка | `def456` |
| `[contenthash]` | Хеш содержимого файла | `ghi789` |
| `[ext]` | Расширение файла | `.js` |
| `[query]` | Строка запроса | `?v=1` |

### Рекомендованная конфигурация для production
```js
output: {
  path: path.resolve(__dirname, 'dist'),
  filename: '[name].[contenthash:8].js',   // JS бандлы
  chunkFilename: '[id].[contenthash:8].js', // lazy-loaded чанки
  assetModuleFilename: 'assets/[hash][ext]', // ресурсы
  publicPath: 'https://cdn.example.com/',   // CDN URL
  clean: true,
}
```

### publicPath — важная настройка
`publicPath` определяет базовый URL, по которому браузер будет загружать ресурсы:

```js
// Для SPA с React Router на корневом пути
publicPath: '/'

// Для приложения на поддомене
publicPath: '/app/'

// Для CDN
publicPath: 'https://cdn.example.com/assets/'

// Автоматически (Webpack 5)
publicPath: 'auto'
```

## context — базовая директория

```js
module.exports = {
  context: path.resolve(__dirname, 'src'),
  entry: './index.js', // относительно context, т.е. src/index.js
};
```

<Playground client:visible template="react" files={{ 'App.tsx': `import { useState } from 'react';

const BG = '#0f172a';
const WEBPACK_COLOR = '#8dd6f9';
const CARD = '#1e293b';
const TEXT = '#e2e8f0';
const ACCENT = '#fb923c';
const SUCCESS = '#4ade80';
const MUTED = '#64748b';

type EntryType = 'string' | 'array' | 'object' | 'descriptor';

interface EntryPoint {
  name: string;
  path: string;
  dependOn?: string;
}

function generateConfig(
  entryType: EntryType,
  entries: EntryPoint[],
  filenamePattern: string,
  publicPath: string,
  clean: boolean,
  useContext: boolean,
  chunkHash: string
): string {
  const lines: string[] = ["const path = require('path');", '', 'module.exports = {'];

  if (useContext) lines.push("  context: path.resolve(__dirname, 'src'),");

  if (entryType === 'string') {
    lines.push(\`  entry: '\${useContext ? './' + entries[0].path.replace('src/', '') : entries[0].path}',\`);
  } else if (entryType === 'array') {
    lines.push("  entry: [");
    lines.push("    './src/polyfills.js',");
    lines.push(\`    '\${entries[0].path}',\`);
    lines.push("  ],");
  } else if (entryType === 'object') {
    lines.push('  entry: {');
    entries.forEach(e => {
      lines.push(\`    \${e.name}: '\${useContext ? './' + e.path.replace('src/', '') : e.path}',\`);
    });
    lines.push('  },');
  } else if (entryType === 'descriptor') {
    lines.push('  entry: {');
    entries.forEach((e, i) => {
      lines.push(\`    \${e.name}: {\`);
      lines.push(\`      import: '\${e.path}',\`);
      if (e.dependOn) lines.push(\`      dependOn: '\${e.dependOn}',\`);
      lines.push(\`      filename: 'pages/\${e.name}.[contenthash:8].js',\`);
      lines.push('    },');
    });
    lines.push('  },');
  }

  lines.push('  output: {');
  lines.push("    path: path.resolve(__dirname, 'dist'),");

  const filename = filenamePattern
    .replace('{hash}', chunkHash === 'contenthash' ? '[contenthash:8]' : chunkHash === 'chunkhash' ? '[chunkhash:8]' : '[hash:8]')
    .replace('{name}', '[name]');

  if (entryType === 'string') {
    lines.push(\`    filename: '\${filename.replace('[name].', '')}',\`);
  } else {
    lines.push(\`    filename: '\${filename}',\`);
    lines.push(\`    chunkFilename: '[id].[contenthash:8].js',\`);
  }

  lines.push(\`    publicPath: '\${publicPath}',\`);
  if (clean) lines.push('    clean: true,');
  lines.push("    assetModuleFilename: 'assets/[hash][ext]',");
  lines.push('  },');
  lines.push('};');

  return lines.join('\\n');
}

const DEFAULT_ENTRIES: EntryPoint[] = [
  { name: 'main', path: './src/index.js' },
  { name: 'admin', path: './src/admin/index.js' },
];

export default function App() {
  const [entryType, setEntryType] = useState<EntryType>('string');
  const [entries, setEntries] = useState<EntryPoint[]>(DEFAULT_ENTRIES);
  const [filenamePattern, setFilenamePattern] = useState('{name}.{hash}.js');
  const [publicPath, setPublicPath] = useState('/');
  const [clean, setClean] = useState(true);
  const [useContext, setUseContext] = useState(false);
  const [chunkHash, setChunkHash] = useState('contenthash');
  const [copied, setCopied] = useState(false);

  const config = generateConfig(entryType, entries, filenamePattern, publicPath, clean, useContext, chunkHash);

  const updateEntry = (index: number, field: keyof EntryPoint, value: string) => {
    const next = [...entries];
    next[index] = { ...next[index], [field]: value };
    setEntries(next);
  };

  const addEntry = () => {
    setEntries([...entries, { name: \`page\${entries.length + 1}\`, path: \`./src/page\${entries.length + 1}/index.js\` }]);
  };

  const removeEntry = (i: number) => {
    if (entries.length > 1) setEntries(entries.filter((_, idx) => idx !== i));
  };

  const copy = () => {
    navigator.clipboard?.writeText(config);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  const entryTypes = [
    { id: 'string', label: 'String', desc: 'Один файл' },
    { id: 'array', label: 'Array', desc: 'Несколько → один бандл' },
    { id: 'object', label: 'Object', desc: 'Несколько бандлов' },
    { id: 'descriptor', label: 'Descriptor', desc: 'Webpack 5 advanced' },
  ] as const;

  const filenamePresets = [
    { label: '[name].[hash].js', value: '{name}.{hash}.js' },
    { label: '[name].bundle.js', value: '{name}.bundle.js' },
    { label: '[name].js', value: '{name}.js' },
  ];

  return (
    <div style={{ background: BG, minHeight: '100vh', color: TEXT, fontFamily: 'monospace', padding: '24px' }}>
      <div style={{ maxWidth: 1000, margin: '0 auto' }}>
        <div style={{ textAlign: 'center', marginBottom: 28 }}>
          <div style={{ fontSize: 26, fontWeight: 700, color: WEBPACK_COLOR, marginBottom: 6 }}>
            🚪 Entry & Output Configurator
          </div>
          <div style={{ color: MUTED, fontSize: 13 }}>Настройте entry/output и получите готовый конфиг</div>
        </div>

        <div style={{ display: 'grid', gridTemplateColumns: '380px 1fr', gap: 24 }}>
          {/* Controls */}
          <div style={{ display: 'flex', flexDirection: 'column', gap: 16 }}>
            {/* Entry type */}
            <div style={{ background: CARD, borderRadius: 12, padding: 16 }}>
              <div style={{ fontSize: 12, color: MUTED, marginBottom: 12, textTransform: 'uppercase' }}>Тип Entry</div>
              <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: 8 }}>
                {entryTypes.map(et => (
                  <div
                    key={et.id}
                    onClick={() => setEntryType(et.id)}
                    style={{
                      padding: '10px', borderRadius: 8, cursor: 'pointer', border: \`2px solid \${entryType === et.id ? WEBPACK_COLOR : 'transparent'}\`,
                      background: entryType === et.id ? '#0f2744' : '#0a0f1e', textAlign: 'center', transition: 'all 0.2s',
                    }}
                  >
                    <div style={{ fontSize: 13, color: entryType === et.id ? WEBPACK_COLOR : TEXT, fontWeight: 700 }}>{et.label}</div>
                    <div style={{ fontSize: 10, color: MUTED, marginTop: 2 }}>{et.desc}</div>
                  </div>
                ))}
              </div>
            </div>

            {/* Entry points */}
            {(entryType === 'object' || entryType === 'descriptor') && (
              <div style={{ background: CARD, borderRadius: 12, padding: 16 }}>
                <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: 12 }}>
                  <div style={{ fontSize: 12, color: MUTED, textTransform: 'uppercase' }}>Entry Points</div>
                  <button onClick={addEntry} style={{ padding: '4px 10px', background: ACCENT, color: BG, border: 'none', borderRadius: 4, cursor: 'pointer', fontSize: 11, fontWeight: 700 }}>+ Добавить</button>
                </div>
                {entries.map((entry, i) => (
                  <div key={i} style={{ background: '#0a0f1e', borderRadius: 8, padding: 10, marginBottom: 8 }}>
                    <div style={{ display: 'flex', gap: 6, marginBottom: 6 }}>
                      <input
                        value={entry.name}
                        onChange={e => updateEntry(i, 'name', e.target.value)}
                        placeholder="name"
                        style={{ flex: 1, background: '#1e293b', border: '1px solid #334155', borderRadius: 4, padding: '4px 8px', color: TEXT, fontSize: 12, fontFamily: 'monospace' }}
                      />
                      <button onClick={() => removeEntry(i)} style={{ padding: '4px 8px', background: '#7f1d1d', color: '#fca5a5', border: 'none', borderRadius: 4, cursor: 'pointer', fontSize: 11 }}>✕</button>
                    </div>
                    <input
                      value={entry.path}
                      onChange={e => updateEntry(i, 'path', e.target.value)}
                      placeholder="./src/index.js"
                      style={{ width: '100%', background: '#1e293b', border: '1px solid #334155', borderRadius: 4, padding: '4px 8px', color: SUCCESS, fontSize: 11, fontFamily: 'monospace', boxSizing: 'border-box' }}
                    />
                  </div>
                ))}
              </div>
            )}

            {/* Output settings */}
            <div style={{ background: CARD, borderRadius: 12, padding: 16 }}>
              <div style={{ fontSize: 12, color: MUTED, marginBottom: 12, textTransform: 'uppercase' }}>Output Settings</div>

              <div style={{ marginBottom: 10 }}>
                <div style={{ fontSize: 11, color: MUTED, marginBottom: 4 }}>filename паттерн</div>
                <div style={{ display: 'flex', gap: 6, flexWrap: 'wrap' }}>
                  {filenamePresets.map(p => (
                    <button
                      key={p.value}
                      onClick={() => setFilenamePattern(p.value)}
                      style={{ padding: '4px 8px', background: filenamePattern === p.value ? WEBPACK_COLOR : '#0a0f1e', color: filenamePattern === p.value ? BG : TEXT, border: 'none', borderRadius: 4, cursor: 'pointer', fontSize: 10, fontFamily: 'monospace' }}
                    >{p.label}</button>
                  ))}
                </div>
              </div>

              <div style={{ marginBottom: 10 }}>
                <div style={{ fontSize: 11, color: MUTED, marginBottom: 4 }}>Hash тип</div>
                <div style={{ display: 'flex', gap: 6 }}>
                  {['contenthash', 'chunkhash', 'hash'].map(h => (
                    <button
                      key={h}
                      onClick={() => setChunkHash(h)}
                      style={{ padding: '4px 8px', background: chunkHash === h ? ACCENT : '#0a0f1e', color: chunkHash === h ? BG : TEXT, border: 'none', borderRadius: 4, cursor: 'pointer', fontSize: 10, fontFamily: 'monospace' }}
                    >{h}</button>
                  ))}
                </div>
              </div>

              <div style={{ marginBottom: 10 }}>
                <div style={{ fontSize: 11, color: MUTED, marginBottom: 4 }}>publicPath</div>
                <input
                  value={publicPath}
                  onChange={e => setPublicPath(e.target.value)}
                  style={{ width: '100%', background: '#0a0f1e', border: '1px solid #334155', borderRadius: 4, padding: '6px 10px', color: SUCCESS, fontSize: 12, fontFamily: 'monospace', boxSizing: 'border-box' }}
                />
              </div>

              <div style={{ display: 'flex', gap: 16 }}>
                <label style={{ display: 'flex', alignItems: 'center', gap: 8, cursor: 'pointer', fontSize: 12 }}>
                  <input type="checkbox" checked={clean} onChange={e => setClean(e.target.checked)} />
                  <span style={{ color: TEXT }}>clean: true</span>
                </label>
                <label style={{ display: 'flex', alignItems: 'center', gap: 8, cursor: 'pointer', fontSize: 12 }}>
                  <input type="checkbox" checked={useContext} onChange={e => setUseContext(e.target.checked)} />
                  <span style={{ color: TEXT }}>context</span>
                </label>
              </div>
            </div>
          </div>

          {/* Config output */}
          <div style={{ background: CARD, borderRadius: 12, padding: 16 }}>
            <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: 12 }}>
              <div style={{ fontSize: 12, color: MUTED, textTransform: 'uppercase' }}>webpack.config.js</div>
              <button
                onClick={copy}
                style={{ padding: '4px 12px', background: copied ? SUCCESS : WEBPACK_COLOR, color: BG, border: 'none', borderRadius: 4, cursor: 'pointer', fontSize: 11, fontFamily: 'monospace', fontWeight: 700 }}
              >{copied ? '✓ Скопировано' : '📋 Копировать'}</button>
            </div>
            <pre style={{ margin: 0, background: '#0a0f1e', borderRadius: 8, padding: '14px', fontSize: 12, lineHeight: 1.7, overflow: 'auto', maxHeight: 560, color: TEXT }}>
              <code>{config}</code>
            </pre>

            {/* Output preview */}
            <div style={{ marginTop: 16 }}>
              <div style={{ fontSize: 12, color: MUTED, marginBottom: 8, textTransform: 'uppercase' }}>Предпросмотр dist/</div>
              <div style={{ background: '#0a0f1e', borderRadius: 8, padding: 12 }}>
                <div style={{ fontSize: 11, color: MUTED, marginBottom: 6 }}>dist/</div>
                {entryType === 'string'
                  ? <div style={{ fontSize: 12, color: SUCCESS, paddingLeft: 16 }}>├── bundle.a1b2c3d4.js</div>
                  : entries.map((e, i) => (
                    <div key={i} style={{ fontSize: 12, color: SUCCESS, paddingLeft: 16 }}>
                      {i === entries.length - 1 ? '└── ' : '├── '}{e.name}.a1b2c3d4.js
                    </div>
                  ))
                }
                <div style={{ fontSize: 12, color: '#a78bfa', paddingLeft: 16 }}>├── styles.css</div>
                <div style={{ fontSize: 12, color: ACCENT, paddingLeft: 16 }}>└── index.html</div>
              </div>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
` }} />
