Git — система контроля версий.

## Основные команды

```bash
git init              # Начать отслеживание
git add .             # Добавить все файлы
git commit -m "msg"   # Сохранить снапшот
git status            # Проверить состояние
git log               # История коммитов
```

## Типичный workflow

```bash
# 1. Внёс изменения в код
# 2. Проверил что изменилось
git status

# 3. Добавил файлы
git add src/

# 4. Закоммитил
git commit -m "feat: добавил новую фичу"
```

## Conventional Commits

- `feat:` — новая фича
- `fix:` — исправление бага
- `docs:` — документация
- `refactor:` — рефакторинг

## Интерактивный пример

Симулятор git init и commit:

<Playground client:visible
  template="vite-react"
  files={{
    "/App.jsx": `import { useState } from "react";

const INIT_FILES = [
  { name: "index.html", status: "untracked" },
  { name: "style.css", status: "untracked" },
  { name: "app.js", status: "untracked" },
];

export default function App() {
  const [initialized, setInitialized] = useState(false);
  const [files, setFiles] = useState(INIT_FILES);
  const [commits, setCommits] = useState([]);
  const [log, setLog] = useState(["Нажми кнопки, чтобы выполнить команды Git"]);

  const addLog = (msg) => setLog((p) => [...p, msg]);

  const gitInit = () => {
    if (initialized) return addLog("Already initialized");
    setInitialized(true);
    addLog("$ git init");
    addLog("Initialized empty Git repository in /project/.git/");
  };

  const gitAdd = () => {
    if (!initialized) return addLog("fatal: not a git repository");
    setFiles((f) => f.map((x) => x.status === "untracked" ? { ...x, status: "staged" } : x));
    addLog("$ git add .");
    addLog("Added all files to staging area");
  };

  const gitCommit = () => {
    if (!initialized) return addLog("fatal: not a git repository");
    const staged = files.filter((f) => f.status === "staged");
    if (!staged.length) return addLog("nothing to commit");
    const msg = "feat: initial commit #" + (commits.length + 1);
    setFiles((f) => f.map((x) => x.status === "staged" ? { ...x, status: "committed" } : x));
    setCommits((c) => [...c, { hash: Math.random().toString(36).slice(2, 9), msg }]);
    addLog("$ git commit -m \\"" + msg + "\\"");
    addLog("[main] " + msg + " (" + staged.length + " files)");
  };

  const colors = { untracked: "#94a3b8", staged: "#4ade80", committed: "#818cf8" };

  return (
    <div style={{ fontFamily: "monospace", background: "#0f172a", color: "#e2e8f0", padding: 16, minHeight: "100vh" }}>
      <h3 style={{ color: "#818cf8", marginBottom: 12 }}>Git Init & Commit</h3>
      <div style={{ display: "flex", gap: 8, marginBottom: 12, flexWrap: "wrap" }}>
        <button onClick={gitInit} style={{ padding: "6px 14px", borderRadius: 6, border: "none", cursor: "pointer", fontWeight: 700, fontSize: 12, background: "#6366f1", color: "#fff" }}>git init</button>
        <button onClick={gitAdd} style={{ padding: "6px 14px", borderRadius: 6, border: "none", cursor: "pointer", fontWeight: 700, fontSize: 12, background: "#22c55e", color: "#0f172a" }}>git add .</button>
        <button onClick={gitCommit} style={{ padding: "6px 14px", borderRadius: 6, border: "none", cursor: "pointer", fontWeight: 700, fontSize: 12, background: "#f59e0b", color: "#0f172a" }}>git commit</button>
      </div>
      <div style={{ display: "flex", gap: 8, marginBottom: 12, flexWrap: "wrap" }}>
        {files.map((f) => (
          <div key={f.name} style={{ background: "#1e293b", border: "1px solid " + colors[f.status], borderRadius: 6, padding: "6px 10px", fontSize: 12 }}>
            <div style={{ color: colors[f.status] }}>{f.name}</div>
            <div style={{ fontSize: 10, color: "#64748b" }}>{f.status}</div>
          </div>
        ))}
      </div>
      {commits.length > 0 && (
        <div style={{ marginBottom: 12 }}>
          <div style={{ fontSize: 11, color: "#64748b", marginBottom: 4 }}>Commits:</div>
          {commits.map((c, i) => (
            <div key={i} style={{ display: "flex", gap: 8, fontSize: 12, padding: "3px 0" }}>
              <span style={{ color: "#f59e0b" }}>{c.hash}</span>
              <span>{c.msg}</span>
            </div>
          ))}
        </div>
      )}
      <div style={{ background: "#1e293b", borderRadius: 8, padding: 10, maxHeight: 120, overflowY: "auto", fontSize: 11, color: "#94a3b8" }}>
        {log.map((l, i) => <div key={i}>{l}</div>)}
      </div>
    </div>
  );
}
`
  }}
/>
