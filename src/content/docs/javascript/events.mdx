## JavaScript: События в браузере и JS

![Иллюстрация к уроку](/lessons/javascript-events.png)


События — это сигналы от браузера о том, что что-то произошло.

## Интерактивная практика: Events

Поэкспериментируй с событиями!
1. Нажми на **Button 1**, чтобы отключить её.
2. Попробуй отправить форму (заметь, что страница не перезагружается благодаря `e.preventDefault()`).
3. Кликни на любой элемент списка (работает через делегирование).

<html>
<head>
  <style>
    body { font-family: sans-serif; padding: 20px; }
    .section { margin-bottom: 20px; padding: 15px; border: 1px solid #ddd; border-radius: 8px; }
    h3 { margin-top: 0; }
    button { padding: 8px 16px; margin-right: 10px; cursor: pointer; }
    button:disabled { background: #ccc; cursor: not-allowed; }
    ul { padding-left: 20px; cursor: pointer; }
    li:hover { background: #f0f0f0; }
    .log { 
      background: #222; color: #0f0; 
      padding: 10px; font-family: monospace; 
      height: 100px; overflow-y: auto; 
      border-radius: 4px;
    }
  </style>
</head>
<body>
  
  <div class="section">
    <h3>1. Кнопки и this</h3>
    <button id="btn1">Нажми меня (Once)</button>
    <button id="btn2">Покажи время</button>
  </div>

  <div class="section">
    <h3>2. Форма (preventDefault)</h3>
    <form id="myForm">
      <input type="text" placeholder="Введите имя..." required>
      <button type="submit">Отправить</button>
    </form>
  </div>

  <div class="section">
    <h3>3. Делегирование (Delegation)</h3>
    <ul id="myList">
      <li>Элемент 1</li>
      <li>Элемент 2</li>
      <li>Элемент 3</li>
      <li>Элемент 4</li>
    </ul>
    <button onclick="addItem()">+ Добавить пункт</button>
  </div>

  <div class="log" id="log">Logs will appear here...</div>

  <script>
    const logDiv = document.getElementById('log');
    function log(msg) {
      logDiv.innerHTML = '> ' + msg + '<br>' + logDiv.innerHTML;
    }

    // 1. Кнопки
    const btn1 = document.getElementById('btn1');
    btn1.addEventListener('click', function(e) {
      log('Button 1 clicked!');
      this.textContent = 'Нажато!';
      this.disabled = true;
      this.style.background = '#e74c3c';
      this.style.color = 'white';
    });

    const btn2 = document.getElementById('btn2');
    btn2.addEventListener('click', (e) => {
      log('Time: ' + new Date().toLocaleTimeString());
      e.stopPropagation(); // Остановить всплытие (если бы было)
    });

    // 2. Форма
    const form = document.getElementById('myForm');
    form.addEventListener('submit', (e) => {
      e.preventDefault(); // ОТМЕНЯЕМ стандартную отправку формы
      const input = form.querySelector('input');
      log('Форма отправлена! Имя: ' + input.value);
      input.value = '';
    });

    // 3. Делегирование
    const list = document.getElementById('myList');
    list.addEventListener('click', (e) => {
      // Проверяем, что клик был по LI
      if (e.target.tagName === 'LI') {
        log('Клик по списку: ' + e.target.textContent);
        e.target.style.textDecoration = 'line-through';
        e.target.style.color = '#888';
      }
    });

    function addItem() {
      const li = document.createElement('li');
      li.textContent = 'Новый элемент ' + (list.children.length + 1);
      list.appendChild(li);
    }
  </script>
</body>
</html>`
