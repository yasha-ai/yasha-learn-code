# Asset Modules — встроенная обработка ресурсов в Webpack 5

**Asset Modules** — это новая встроенная система Webpack 5 для обработки статических ресурсов: изображений, шрифтов, SVG, медиафайлов. Она заменяет устаревшие лоадеры `url-loader`, `file-loader` и `raw-loader` без дополнительных зависимостей.

## Устаревший подход (Webpack 4)

```bash
# Нужно было устанавливать отдельно
npm install --save-dev file-loader url-loader raw-loader
```

```js
// webpack.config.js (старый способ)
module: {
  rules: [
    { test: /\\.(png|jpg|gif)$/, use: 'file-loader' },
    { test: /\\.(png|jpg|gif)$/, use: { loader: 'url-loader', options: { limit: 8192 } } },
    { test: /\\.txt$/, use: 'raw-loader' },
  ]
}
```

## Asset Modules в Webpack 5

```js
// webpack.config.js (новый способ — без npm install!)
module: {
  rules: [
    { test: /\\.(png|jpg|gif|svg)$/, type: 'asset/resource' },
    { test: /\\.svg$/, type: 'asset/inline' },
    { test: /\\.txt$/, type: 'asset/source' },
    { test: /\\.(png|jpg)$/, type: 'asset' }, // автоматический выбор
  ]
}
```

## Четыре типа Asset Modules

### 1. `asset/resource` — замена file-loader

Копирует файл в папку output и возвращает **URL**:

```js
{ test: /\\.(png|jpg|gif|webp)$/, type: 'asset/resource' }

// Настройка имени файла:
{
  test: /\\.(png|jpg|gif)$/,
  type: 'asset/resource',
  generator: {
    filename: 'images/[hash][ext][query]',
  },
}
```

```js
import logo from './logo.png';
// logo === '/dist/images/abc123.png'

// В JSX:
<img src={logo} alt="Logo" />
```

### 2. `asset/inline` — замена url-loader

Конвертирует файл в **base64 Data URL** и инлайнит в бандл:

```js
{
  test: /\\.(png|jpg|gif)$/,
  type: 'asset/inline',
}
```

```js
import logo from './logo.png';
// logo === 'data:image/png;base64,iVBORw0KGgo...'

// Плюсы: нет отдельного HTTP запроса
// Минусы: увеличивает размер JS бандла
// Используйте только для маленьких файлов (<8 KB)
```

### 3. `asset/source` — замена raw-loader

Возвращает **строку** с содержимым файла:

```js
{ test: /\\.txt$/, type: 'asset/source' }
{ test: /\\.svg$/, type: 'asset/source' } // SVG как строка
{ test: /\\.graphql$/, type: 'asset/source' } // GraphQL запросы
```

```js
import content from './readme.txt';
// content === 'Hello, World!\\nThis is a text file.'

import svgMarkup from './icon.svg';
// svgMarkup === '<svg xmlns="..."><path d="..."/></svg>'
```

### 4. `asset` — умный выбор

Автоматически выбирает между `resource` и `inline` на основе размера файла:

```js
{
  test: /\\.(png|jpg|gif)$/,
  type: 'asset',
  parser: {
    dataUrlCondition: {
      maxSize: 8 * 1024, // 8 KB — меньше → inline, больше → resource
    },
  },
}
```

## Глобальная настройка имён файлов

```js
output: {
  assetModuleFilename: 'assets/[hash][ext][query]',
  // → dist/assets/a1b2c3d4.png
}
```

## Примеры для разных типов файлов

```js
module: {
  rules: [
    // Изображения — умный выбор (< 8KB → inline, > 8KB → file)
    {
      test: /\\.(png|jpg|gif|webp|avif)$/i,
      type: 'asset',
      parser: { dataUrlCondition: { maxSize: 8192 } },
      generator: { filename: 'images/[hash:8][ext]' },
    },

    // SVG — всегда как строка (для React компонентов)
    // или resource (для <img>)
    {
      test: /\\.svg$/,
      type: 'asset/resource',
      generator: { filename: 'icons/[hash:8][ext]' },
    },

    // Шрифты — всегда как файл
    {
      test: /\\.(woff|woff2|eot|ttf|otf)$/i,
      type: 'asset/resource',
      generator: { filename: 'fonts/[name][ext]' },
    },

    // Видео и аудио
    {
      test: /\\.(mp4|webm|ogg|mp3|wav)$/i,
      type: 'asset/resource',
      generator: { filename: 'media/[hash:8][ext]' },
    },
  ],
}
```

const ACCENT = '#fb923c';
const SUCCESS = '#4ade80';
const MUTED = '#64748b';
const PURPLE = '#a78bfa';

type AssetType = 'resource' | 'inline' | 'source' | 'auto';

interface AssetFile {
  name: string;
  ext: string;
  size: number; // KB
  icon: string;
  description: string;

const ASSET_FILES: AssetFile[] = [

  { name: 'avatar', ext: 'jpg', size: 12.1, icon: '👤', description: 'Аватар пользователя' },

}> = {
  resource: {

    color: WEBPACK_COLOR,
    desc: 'Всегда копирует файл в dist/, возвращает URL',

      type: 'URL string',
      value: \`"/dist/assets/\${file.name}.a1b2c3d4.\${file.ext}"\`,
      color: WEBPACK_COLOR,
    }),
  },
  inline: {
    label: 'asset/inline',
    color: PURPLE,

      color: PURPLE,
    }),
  },
  source: {
    label: 'asset/source',
    color: SUCCESS,
    desc: 'Возвращает содержимое файла как строку',

    }),
  },
  auto: {
    label: 'asset (auto)',
    color: ACCENT,
    desc: 'Умный выбор: маленький файл → inline, большой → resource',
    replacedLoader: 'url-loader (с limit)',
    getResult: (file, maxSize) => {
      const isInline = file.size <= maxSize;
      return {
        type: isInline ? 'Data URL (inline)' : 'URL string (file)',
        value: isInline
          ? \`"data:image/\${file.ext};base64,iVBORw0..."\`
          : \`"/dist/assets/\${file.name}.\${file.ext}"\`,
        color: isInline ? PURPLE : WEBPACK_COLOR,
      };
    },
  },
};

function generateRule(type: AssetType, ext: string, maxSize: number): string {
  const lines = ['{'];
  lines.push(\`  test: /\\\\.\${ext}$/i,\`);

  if (type === 'auto') {
    lines.push("  type: 'asset',");
    lines.push('  parser: {');
    lines.push('    dataUrlCondition: {');
    lines.push(\`      maxSize: \${maxSize * 1024}, // \${maxSize} KB\`);
    lines.push('    },');
    lines.push('  },');
  } else {
    lines.push(\`  type: 'asset/\${type}',\`);
  }

  if (type === 'resource' || type === 'auto') {
    lines.push('  generator: {');
    lines.push(\`    filename: 'assets/[hash:8][ext]',\`);
    lines.push('  },');
  }

  lines.push('}');
  return lines.join('\\n');
}

            >
              <div style={{ fontSize: 11, color: selectedType === key ? val.color : TEXT, fontWeight: 700, marginBottom: 4 }}>{val.label}</div>
              <div style={{ fontSize: 9, color: MUTED }}>replaces: {val.replacedLoader}</div>
            </div>
          ))}
        </div>

        <div style={{ display: 'grid', gridTemplateColumns: '280px 1fr', gap: 24 }}>
          {/* File selector */}
          <div style={{ display: 'flex', flexDirection: 'column', gap: 12 }}>
            <div style={{ background: CARD, borderRadius: 12, padding: 16 }}>
              <div style={{ fontSize: 12, color: MUTED, marginBottom: 12, textTransform: 'uppercase' }}>Выберите файл</div>

                <div

'#0f2744' : '#0a0f1e', border: \`1px solid \${selectedFile === f.name ? typeInfo.color : 'transparent'}\`, marginBottom: 6, transition: 'all 0.2s' }}
