# Тестирование с Vitest

Vitest — это современный тест-раннер, созданный специально для Vite-проектов. Он использует ту же конфигурацию `vite.config.ts`, поддерживает ESM из коробки и работает значительно быстрее Jest благодаря нативной интеграции с Vite.

## Установка и настройка

```bash
npm i -D vitest
# или с jsdom для тестирования DOM
npm i -D vitest jsdom @testing-library/react @testing-library/jest-dom
```

## Конфигурация в vite.config.ts

```typescript
// vite.config.ts
import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'

export default defineConfig({
  plugins: [react()],
  test: {
    // Среда выполнения тестов
    environment: 'jsdom', // 'node' | 'jsdom' | 'happy-dom'

    // Глобальные функции (describe, it, expect без импорта)
    globals: true,

    // Setup файл (импорт jest-dom матчеров)
    setupFiles: ['./src/setupTests.ts'],

    // Coverage
    coverage: {
      provider: 'v8',   // 'v8' | 'istanbul'
      reporter: ['text', 'html', 'lcov'],
      include: ['src/**'],
      exclude: ['src/**/*.stories.tsx'],
      thresholds: {
        lines: 80,
        functions: 80,
        branches: 70,
      },
    },

    // Параллельное выполнение
    pool: 'threads',    // 'threads' | 'forks' | 'vmThreads'
    poolOptions: {
      threads: {
        singleThread: false,
      },
    },
  },
})
```

## Написание тестов

```typescript
// src/utils/math.test.ts
import { describe, it, expect } from 'vitest'
import { add, multiply, factorial } from './math'

describe('math utils', () => {
  describe('add()', () => {
    it('складывает два числа', () => {
      expect(add(2, 3)).toBe(5)
    })

    it('работает с отрицательными числами', () => {
      expect(add(-1, 1)).toBe(0)
    })
  })

  it('factorial(5) === 120', () => {
    expect(factorial(5)).toBe(120)
  })
})
```

## Тестирование React компонентов

```typescript
// src/components/Button.test.tsx
import { render, screen, fireEvent } from '@testing-library/react'
import { describe, it, expect, vi } from 'vitest'
import { Button } from './Button'

describe('Button', () => {
  it('отображает текст', () => {
    render(<Button>Click me</Button>)
    expect(screen.getByText('Click me')).toBeInTheDocument()
  })

  it('вызывает onClick при клике', () => {
    const handleClick = vi.fn()
    render(<Button onClick={handleClick}>Click me</Button>)
    fireEvent.click(screen.getByText('Click me'))
    expect(handleClick).toHaveBeenCalledOnce()
  })

  it('disabled кнопка не реагирует на клики', () => {
    const handleClick = vi.fn()
    render(<Button disabled onClick={handleClick}>Disabled</Button>)
    fireEvent.click(screen.getByText('Disabled'))
    expect(handleClick).not.toHaveBeenCalled()
  })
})
```

## Vitest специфичные возможности

```typescript
// Mock функции
import { vi } from 'vitest'

// Мок модуля
vi.mock('./api', () => ({
  fetchUser: vi.fn().mockResolvedValue({ id: 1, name: 'Alice' }),
}))

// Шпион
const spy = vi.spyOn(console, 'error')
expect(spy).toHaveBeenCalledWith('Expected error')

// Таймеры
vi.useFakeTimers()
vi.advanceTimersByTime(1000)
vi.useRealTimers()

// Инлайн снапшоты
expect(result).toMatchInlineSnapshot(\`
  {
    "name": "Alice",
    "age": 30,
  }
\`)
```

## CLI команды

```bash
# Запуск всех тестов
npx vitest

# Watch mode (для разработки)
npx vitest --watch

# UI mode — красивый веб-интерфейс
npx vitest --ui

# Coverage
npx vitest run --coverage

# Конкретный файл
npx vitest Button.test.tsx

# Bench (бенчмарки производительности)
npx vitest bench
```

## Vitest UI

```bash
npm i -D @vitest/ui
npx vitest --ui
# Открывает http://localhost:51204/__vitest__/
```

<Playground template="react" files={{ 'App.tsx': `import { useState, useRef } from 'react';

const BG = '#0f172a';
const VITE_COLOR = '#646cff';
const CARD = '#1e293b';
const TEXT = '#e2e8f0';
const ACCENT = '#38bdf8';
const GREEN = '#4ade80';
const RED = '#f87171';
const AMBER = '#f59e0b';

interface TestCase {
  id: string;
  suite: string;
  name: string;
  status: 'pending' | 'running' | 'pass' | 'fail' | 'skip';
  duration?: number;
  error?: string;
}

const initialTests: TestCase[] = [
  { id: 't1', suite: 'math utils > add()', name: 'складывает два числа', status: 'pending' },
  { id: 't2', suite: 'math utils > add()', name: 'работает с отрицательными', status: 'pending' },
  { id: 't3', suite: 'math utils', name: 'factorial(5) === 120', status: 'pending' },
  { id: 't4', suite: 'Button', name: 'отображает текст', status: 'pending' },
  { id: 't5', suite: 'Button', name: 'вызывает onClick при клике', status: 'pending' },
  { id: 't6', suite: 'Button', name: 'disabled не реагирует на клики', status: 'pending' },
  { id: 't7', suite: 'useCounter hook', name: 'начальное значение 0', status: 'pending' },
  { id: 't8', suite: 'useCounter hook', name: 'increment увеличивает на 1', status: 'pending' },
  { id: 't9', suite: 'useCounter hook', name: 'decrement уменьшает на 1', status: 'pending' },
  { id: 't10', suite: 'API utils', name: 'fetchUser возвращает пользователя', status: 'pending' },
  { id: 't11', suite: 'API utils', name: 'обрабатывает 404 ошибку', status: 'pending', error: "Expected status 404, got 200" },
  { id: 't12', suite: 'Router', name: 'рендерит HomePage на /', status: 'pending' },
];

const outcomes: Record<string, 'pass' | 'fail'> = {
  t1: 'pass', t2: 'pass', t3: 'pass',
  t4: 'pass', t5: 'pass', t6: 'pass',
  t7: 'pass', t8: 'pass', t9: 'pass',
  t10: 'pass', t11: 'fail', t12: 'pass',
};

const durations: Record<string, number> = {
  t1: 2, t2: 1, t3: 3, t4: 8, t5: 12, t6: 11,
  t7: 4, t8: 5, t9: 4, t10: 24, t11: 18, t12: 15,
};

export default function App() {
  const [tests, setTests] = useState<TestCase[]>(initialTests);
  const [running, setRunning] = useState(false);
  const [finished, setFinished] = useState(false);
  const [filter, setFilter] = useState<'all' | 'pass' | 'fail' | 'pending'>('all');
  const [elapsed, setElapsed] = useState(0);
  const startRef = useRef<number>(0);

  const runTests = async () => {
    setRunning(true);
    setFinished(false);
    setElapsed(0);
    startRef.current = Date.now();

    const reset = initialTests.map(t => ({ ...t, status: 'pending' as const, duration: undefined, error: undefined }));
    setTests(reset);

    await new Promise(r => setTimeout(r, 50));

    for (let i = 0; i < initialTests.length; i++) {
      const test = initialTests[i];
      setTests(prev => prev.map(t => t.id === test.id ? { ...t, status: 'running' } : t));
      await new Promise(r => setTimeout(r, 150 + Math.random() * 100));
      const outcome = outcomes[test.id];
      setTests(prev => prev.map(t => t.id === test.id ? {
        ...t,
        status: outcome,
        duration: durations[test.id],
        error: outcome === 'fail' ? t.error : undefined,
      } : t));
      setElapsed(Date.now() - startRef.current);
    }

    setRunning(false);
    setFinished(true);
    setElapsed(Date.now() - startRef.current);
  };

  const reset = () => {
    setTests(initialTests);
    setRunning(false);
    setFinished(false);
    setElapsed(0);
  };

  const suites = Array.from(new Set(tests.map(t => t.suite)));

  const passed = tests.filter(t => t.status === 'pass').length;
  const failed = tests.filter(t => t.status === 'fail').length;
  const pending = tests.filter(t => t.status === 'pending').length;
  const running_count = tests.filter(t => t.status === 'running').length;

  const filtered = filter === 'all' ? tests
    : filter === 'pass' ? tests.filter(t => t.status === 'pass')
    : filter === 'fail' ? tests.filter(t => t.status === 'fail')
    : tests.filter(t => t.status === 'pending');

  const statusIcon = (s: TestCase['status']) => ({
    pending: <span style={{ color: '#64748b', fontSize: 12 }}>○</span>,
    running: <span style={{ color: AMBER, fontSize: 12 }}>⟳</span>,
    pass: <span style={{ color: GREEN, fontSize: 12 }}>✓</span>,
    fail: <span style={{ color: RED, fontSize: 12 }}>✗</span>,
    skip: <span style={{ color: '#64748b', fontSize: 12 }}>⊘</span>,
  })[s];

  return (
    <div style={{ background: BG, minHeight: '100vh', padding: '24px 16px', fontFamily: 'system-ui, sans-serif', color: TEXT }}>
      <div style={{ maxWidth: 680, margin: '0 auto' }}>
        {/* Header */}
        <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: 24 }}>
          <div>
            <h1 style={{
              margin: 0, fontSize: 20, fontWeight: 800,
              background: \`linear-gradient(90deg, \${VITE_COLOR}, \${ACCENT})\`,
              WebkitBackgroundClip: 'text', WebkitTextFillColor: 'transparent',
            }}>⚡ Vitest Runner</h1>
            <div style={{ fontSize: 12, color: '#64748b', marginTop: 3 }}>Симулятор тест-раннера</div>
          </div>
          <div style={{ fontFamily: 'monospace', fontSize: 12, color: '#64748b' }}>
            {finished ? \`\${elapsed}ms\` : running ? \`\${elapsed}ms...\` : '—'}
          </div>
        </div>

        {/* Stats */}
        <div style={{ display: 'flex', gap: 8, marginBottom: 20 }}>
          {[
            { label: 'Всего', value: tests.length, color: TEXT },
            { label: 'Passed ✓', value: passed, color: GREEN },
            { label: 'Failed ✗', value: failed, color: RED },
            { label: 'Pending', value: pending, color: '#64748b' },
          ].map(s => (
            <div key={s.label} style={{ flex: 1, background: CARD, borderRadius: 10, padding: '10px 6px', textAlign: 'center', border: '1px solid #334155' }}>
              <div style={{ fontSize: 20, fontWeight: 800, color: s.color }}>{s.value}</div>
              <div style={{ fontSize: 10, color: '#64748b' }}>{s.label}</div>
            </div>
          ))}
        </div>

        {/* Progress bar */}
        {(running || finished) && (
          <div style={{ height: 6, background: '#334155', borderRadius: 3, marginBottom: 16, overflow: 'hidden' }}>
            <div style={{
              height: '100%',
              width: \`\${((passed + failed) / tests.length) * 100}%\`,
              background: failed > 0 ? RED : GREEN, borderRadius: 3, transition: 'width 0.2s',
            }} />
          </div>
        )}

        {/* Controls */}
        <div style={{ display: 'flex', gap: 8, marginBottom: 16 }}>
          <button onClick={runTests} disabled={running} style={{
            flex: 2, padding: '11px 0', borderRadius: 10, border: 'none', cursor: running ? 'not-allowed' : 'pointer',
            background: running ? '#334155' : VITE_COLOR, color: '#fff', fontWeight: 700, fontSize: 14,
          }}>
            {running ? \`⏳ Запуск тестов... (\${passed + failed}/\${tests.length})\` : '▶ Запустить тесты'}
          </button>
          <button onClick={reset} style={{
            flex: 1, padding: '11px 0', borderRadius: 10, border: '1px solid #334155', cursor: 'pointer',
            background: 'transparent', color: TEXT, fontSize: 13,
          }}>Сброс</button>
        </div>

        {/* Filter */}
        <div style={{ display: 'flex', gap: 6, marginBottom: 14 }}>
          {(['all', 'pass', 'fail', 'pending'] as const).map(f => (
            <button key={f} onClick={() => setFilter(f)} style={{
              padding: '4px 12px', borderRadius: 14, border: 'none', cursor: 'pointer',
              background: filter === f ? (f === 'pass' ? GREEN : f === 'fail' ? RED : VITE_COLOR) : CARD,
              color: filter === f ? (f === 'pass' || f === 'fail' ? BG : '#fff') : '#94a3b8',
              fontSize: 11, fontWeight: filter === f ? 700 : 400,
            }}>
              {f === 'all' ? \`All (\${tests.length})\` : f === 'pass' ? \`✓ \${passed}\` : f === 'fail' ? \`✗ \${failed}\` : \`Pending \${pending}\`}
            </button>
          ))}
        </div>

        {/* Test list by suite */}
        {suites.filter(s => filtered.some(t => t.suite === s)).map(suite => (
          <div key={suite} style={{ background: CARD, borderRadius: 12, marginBottom: 10, overflow: 'hidden', border: '1px solid #334155' }}>
            <div style={{ padding: '8px 14px', borderBottom: '1px solid #334155', fontSize: 12, color: ACCENT, fontWeight: 700 }}>
              📦 {suite}
            </div>
            {filtered.filter(t => t.suite === suite).map(test => (
              <div key={test.id} style={{
                display: 'flex', justifyContent: 'space-between', alignItems: 'flex-start',
                padding: '8px 14px', borderBottom: '1px solid #1a1a2e',
              }}>
                <div style={{ display: 'flex', alignItems: 'flex-start', gap: 8 }}>
                  <div style={{ marginTop: 1 }}>{statusIcon(test.status)}</div>
                  <div>
                    <div style={{
                      fontSize: 13,
                      color: test.status === 'pass' ? TEXT : test.status === 'fail' ? RED : '#64748b',
                    }}>
                      {test.name}
                    </div>
                    {test.error && (
                      <div style={{ fontSize: 11, color: RED, marginTop: 3, fontFamily: 'monospace' }}>
                        {test.error}
                      </div>
                    )}
                  </div>
                </div>
                {test.duration && (
                  <span style={{ fontSize: 11, color: '#475569', fontFamily: 'monospace', flexShrink: 0 }}>
                    {test.duration}ms
                  </span>
                )}
              </div>
            ))}
          </div>
        ))}

        {finished && (
          <div style={{
            textAlign: 'center', padding: 16, borderRadius: 12, marginTop: 8,
            background: failed > 0 ? \`\${RED}10\` : \`\${GREEN}10\`,
            border: \`1px solid \${failed > 0 ? RED : GREEN}30\`,
          }}>
            <div style={{ fontSize: 20, marginBottom: 4 }}>{failed > 0 ? '❌' : '✅'}</div>
            <div style={{ fontWeight: 700, color: failed > 0 ? RED : GREEN }}>
              {failed > 0 ? \`\${failed} test(s) failed\` : 'All tests passed!'}
            </div>
            <div style={{ fontSize: 12, color: '#64748b', marginTop: 4 }}>
              {passed} passed, {failed} failed — {elapsed}ms
            </div>
          </div>
        )}
      </div>
    </div>
  );
}` }} />
