# vite.config.ts: Настройка конфигурации

Конфигурационный файл Vite — `vite.config.ts` — это точка входа для всех настроек сборщика. Он экспортирует объект конфигурации, обёрнутый в хелпер `defineConfig` для TypeScript-автодополнения.

## Базовая структура

```typescript
// vite.config.ts
import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'
import path from 'path'

export default defineConfig({
  // Плагины
  plugins: [react()],

  // Разрешение модулей
  resolve: {
    alias: {
      '@': path.resolve(__dirname, './src'),
      '@components': path.resolve(__dirname, './src/components'),
    },
  },

  // Настройки dev сервера
  server: {
    port: 3000,
    open: true,
    proxy: {
      '/api': 'http://localhost:8080',
    },
  },

  // Настройки production сборки
  build: {
    outDir: 'dist',
    sourcemap: true,
    target: 'es2015',
  },

  // Базовый URL (для деплоя на поддиректорию)
  base: '/',
})
```

## defineConfig — Зачем нужен?

```typescript
// Без defineConfig — работает, но нет TypeScript-типов
export default {
  plugins: [react()],
  // ❌ нет автодополнения, нет проверки типов
}

// С defineConfig — полная TypeScript поддержка
export default defineConfig({
  plugins: [react()],
  // ✓ IntelliSense, автодополнение, проверка типов
})
```

## resolve.alias — Упрощение импортов

```typescript
// vite.config.ts
resolve: {
  alias: {
    '@': path.resolve(__dirname, './src'),
  },
}

// tsconfig.json — нужно добавить и сюда!
{
  "compilerOptions": {
    "paths": {
      "@/*": ["./src/*"]
    }
  }
}

// Теперь вместо:
import Button from '../../../components/Button'

// Можно писать:
import Button from '@/components/Button'
```

## build опции

```typescript
build: {
  outDir: 'dist',          // Директория вывода
  assetsDir: 'assets',     // Поддиректория для ресурсов
  sourcemap: true,         // Генерировать source maps
  minify: 'esbuild',       // 'esbuild' | 'terser' | false
  target: 'es2015',        // Target браузеры/версии

  // Rollup опции
  rollupOptions: {
    output: {
      manualChunks: {
        vendor: ['react', 'react-dom'],
        utils: ['lodash', 'date-fns'],
      },
    },
  },

  // Лимит для inline (base64) ресурсов
  assetsInlineLimit: 4096, // 4kb

  // Разбивка CSS в отдельные файлы
  cssCodeSplit: true,
}
```

## base URL

```typescript
// Для деплоя на GitHub Pages (например /my-repo/)
base: '/my-repo/',

// Для деплоя в корень
base: '/',

// Из переменной окружения
base: process.env.NODE_ENV === 'production' ? '/app/' : '/',
```

## Conditional Config с mode

```typescript
export default defineConfig(({ command, mode }) => {
  // command: 'serve' (dev) | 'build' (prod)
  // mode: 'development' | 'production' | custom

  return {
    plugins: [
      react(),
      ...(mode === 'development' ? [devPlugin()] : []),
    ],
    build: {
      sourcemap: command === 'serve',
    },
  }
})
```

function generateConfig(options: ConfigOption[]): string {
  const enabled = options.filter(o => o.enabled);
  const hasAlias = enabled.some(o => o.id === 'alias');
  const hasProxy = enabled.some(o => o.id === 'proxy');
  const hasPort = enabled.some(o => o.id === 'server-port');

  let lines: string[] = [];

  // Imports
  lines.push("import { defineConfig } from 'vite'");
  if (enabled.some(o => o.id === 'react-plugin')) {
    lines.push("import react from '@vitejs/plugin-react'");
  }
  if (hasAlias) {
    lines.push("import path from 'path'");
  }
  lines.push('');
  lines.push('export default defineConfig({');

  // Plugins
  if (enabled.some(o => o.id === 'react-plugin')) {
    lines.push('  plugins: [react()],');
    lines.push('');
  }

  // Resolve
  if (hasAlias) {
    lines.push('  resolve: {');
    lines.push('    alias: {');
    lines.push("      '@': path.resolve(__dirname, './src'),");
    lines.push('    },');
    lines.push('  },');
    lines.push('');
  }

  // Server
  const serverOpts = [];
  if (hasPort) serverOpts.push('    port: 3000,', '    open: true,');
  if (hasProxy) serverOpts.push("    proxy: { '/api': { target: 'http://localhost:8080', changeOrigin: true } },");
  if (serverOpts.length) {
    lines.push('  server: {');
    serverOpts.forEach(l => lines.push(l));
    lines.push('  },');
    lines.push('');
  }

  // Build
  const hasSM = enabled.some(o => o.id === 'sourcemap');
  const hasMC = enabled.some(o => o.id === 'manual-chunks');
  if (hasSM || hasMC) {
    lines.push('  build: {');
    if (hasSM) lines.push('    sourcemap: true,');
    if (hasMC) {
      lines.push('    rollupOptions: {');
      lines.push('      output: {');
      lines.push('        manualChunks: {');
      lines.push("          vendor: ['react', 'react-dom'],");
      lines.push('        },');
      lines.push('      },');
      lines.push('    },');
    }
    lines.push('  },');
    lines.push('');
  }

  // General
  if (enabled.some(o => o.id === 'base')) {
    lines.push("  base: process.env.VITE_BASE_URL ?? '/',");
  }
  if (enabled.some(o => o.id === 'env-prefix')) {
    lines.push("  envPrefix: 'APP_',");
  }

  lines.push('})');
  return lines.join('\\n');
