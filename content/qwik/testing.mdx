# Тестирование Qwik приложений

Qwik City рекомендует использовать **Vitest** для unit-тестов и тестирования компонентов, и **Playwright** для E2E тестирования.

## Настройка Vitest

```bash
npm run qwik add vitest
```

```typescript
// vitest.config.ts
import { defineConfig } from 'vitest/config';
import { qwikVite } from '@builder.io/qwik/optimizer';

export default defineConfig({
  plugins: [qwikVite()],
  test: {
    globals: true,
    environment: 'node',
  },
});
```

## Unit тестирование функций

```typescript
// src/utils/format.ts
export function formatPrice(price: number, currency = 'RUB'): string {
  return new Intl.NumberFormat('ru-RU', {
    style: 'currency',
    currency,
  }).format(price);
}

// src/utils/format.test.ts
import { describe, it, expect } from 'vitest';
import { formatPrice } from './format';

describe('formatPrice', () => {
  it('форматирует рублёвую цену', () => {
    expect(formatPrice(1000)).toBe('1 000 ₽');
  });
  
  it('форматирует цену в долларах', () => {
    expect(formatPrice(99.99, 'USD')).toContain('99.99');
  });
  
  it('обрабатывает ноль', () => {
    expect(formatPrice(0)).toBe('0 ₽');
  });
});
```

## Тестирование компонентов

Qwik предоставляет `createDOM` для тестирования компонентов без браузера:

```typescript
// src/components/counter.test.ts
import { createDOM } from '@builder.io/qwik/testing';
import { describe, it, expect } from 'vitest';
import { Counter } from './counter';

describe('Counter', () => {
  it('рендерит начальное значение', async () => {
    const { screen, render } = await createDOM();
    
    await render(<Counter initialValue={5} />);
    
    expect(screen.querySelector('[data-test="count"]')?.textContent).toBe('5');
  });
  
  it('увеличивает счётчик при клике', async () => {
    const { screen, render, userEvent } = await createDOM();
    
    await render(<Counter initialValue={0} />);
    
    const button = screen.querySelector('[data-test="increment"]')!;
    await userEvent(button, 'click');
    
    expect(screen.querySelector('[data-test="count"]')?.textContent).toBe('1');
  });
});
```

## Тестирование routeLoader$

```typescript
// src/routes/blog/[slug]/index.test.ts
import { describe, it, expect, vi } from 'vitest';
import { createRequestEvent } from '@builder.io/qwik-city/testing';

// Мокаем БД
vi.mock('../../../lib/db', () => ({
  db: {
    posts: {
      findUnique: vi.fn(),
    },
  },
}));

describe('usePost loader', () => {
  it('возвращает пост по slug', async () => {
    const mockPost = { id: '1', title: 'Test', slug: 'test' };
    vi.mocked(db.posts.findUnique).mockResolvedValue(mockPost);
    
    const event = createRequestEvent({ params: { slug: 'test' } });
    const result = await usePost.call(event);
    
    expect(result).toEqual(mockPost);
  });
  
  it('возвращает 404 если пост не найден', async () => {
    vi.mocked(db.posts.findUnique).mockResolvedValue(null);
    
    const event = createRequestEvent({ params: { slug: 'nonexistent' } });
    
    await expect(usePost.call(event)).rejects.toThrow();
  });
});
```

## E2E тестирование с Playwright

```bash
npm run qwik add playwright
```

```typescript
// tests/home.spec.ts
import { test, expect } from '@playwright/test';

test('главная страница загружается', async ({ page }) => {
  await page.goto('/');
  
  await expect(page).toHaveTitle(/Мой Qwik App/);
  await expect(page.locator('h1')).toBeVisible();
});

test('счётчик работает', async ({ page }) => {
  await page.goto('/counter');
  
  const count = page.locator('[data-test="count"]');
  const increment = page.locator('[data-test="increment"]');
  
  await expect(count).toHaveText('0');
  await increment.click();
  await expect(count).toHaveText('1');
});

test('форма отправляется', async ({ page }) => {
  await page.goto('/contact');
  
  await page.fill('[name="email"]', 'test@example.com');
  await page.fill('[name="message"]', 'Привет!');
  await page.click('[type="submit"]');
  
  await expect(page.locator('.success-message')).toBeVisible();
});
```

## Playground: Симулятор тестов

<Playground template="react" files={{ 'App.tsx': `import { useState, useCallback } from 'react';

const BG = '#0f172a';
const QWIK = '#ac7ef4';
const CARD = '#1e293b';
const TEXT = '#e2e8f0';
const MUTED = '#64748b';
const GREEN = '#22c55e';
const RED = '#ef4444';
const ORANGE = '#f97316';
const BLUE = '#38bdf8';

type TestStatus = 'pending' | 'running' | 'passed' | 'failed' | 'skipped';

type TestCase = {
  id: string;
  name: string;
  suite: string;
  type: 'unit' | 'component' | 'e2e';
  status: TestStatus;
  duration?: number;
  error?: string;
};

const allTests: TestCase[] = [
  // Unit tests
  { id: 'u1', name: 'форматирует рублёвую цену', suite: 'formatPrice', type: 'unit', status: 'pending' },
  { id: 'u2', name: 'форматирует USD цену', suite: 'formatPrice', type: 'unit', status: 'pending' },
  { id: 'u3', name: 'обрабатывает ноль', suite: 'formatPrice', type: 'unit', status: 'pending' },
  { id: 'u4', name: 'валидирует email корректно', suite: 'validateEmail', type: 'unit', status: 'pending' },
  { id: 'u5', name: 'отклоняет неверный email', suite: 'validateEmail', type: 'unit', status: 'pending' },

  // Component tests
  { id: 'c1', name: 'рендерит начальное значение', suite: 'Counter', type: 'component', status: 'pending' },
  { id: 'c2', name: 'увеличивает при клике', suite: 'Counter', type: 'component', status: 'pending' },
  { id: 'c3', name: 'уменьшает при клике', suite: 'Counter', type: 'component', status: 'pending' },
  { id: 'c4', name: 'рендерит кнопку с label', suite: 'Button', type: 'component', status: 'pending' },
  { id: 'c5', name: 'применяет variant класс', suite: 'Button', type: 'component', status: 'pending' },
  { id: 'c6', name: 'вызывает onClick$ при клике', suite: 'Button', type: 'component', status: 'pending' },

  // E2E tests
  { id: 'e1', name: 'главная страница загружается', suite: 'Home', type: 'e2e', status: 'pending' },
  { id: 'e2', name: 'навигация работает', suite: 'Navigation', type: 'e2e', status: 'pending' },
  { id: 'e3', name: 'счётчик работает', suite: 'Counter E2E', type: 'e2e', status: 'pending' },
  { id: 'e4', name: 'форма отправляется успешно', suite: 'Form E2E', type: 'e2e', status: 'pending' },
  { id: 'e5', name: 'форма показывает ошибки', suite: 'Form E2E', type: 'e2e', status: 'pending' },
];

const typeColors: Record<string, string> = { unit: BLUE, component: QWIK, e2e: ORANGE };
const typeLabels: Record<string, string> = { unit: 'Unit', component: 'Component', e2e: 'E2E' };

const statusColors: Record<TestStatus, string> = {
  pending: MUTED, running: ORANGE, passed: GREEN, failed: RED, skipped: '#64748b',
};

const statusIcons: Record<TestStatus, string> = {
  pending: '○', running: '⟳', passed: '✓', failed: '✕', skipped: '–',
};

export default function App() {
  const [tests, setTests] = useState(allTests);
  const [isRunning, setIsRunning] = useState(false);
  const [filter, setFilter] = useState<'all' | 'unit' | 'component' | 'e2e'>('all');
  const [statusFilter, setStatusFilter] = useState<TestStatus | 'all'>('all');

  const runTests = useCallback(async (filterType?: string) => {
    const toRun = filterType ? allTests.filter(t => t.type === filterType) : allTests;
    setIsRunning(true);
    setTests(prev => prev.map(t => toRun.find(r => r.id === t.id) ? { ...t, status: 'pending', duration: undefined, error: undefined } : t));

    for (const test of toRun) {
      setTests(prev => prev.map(t => t.id === test.id ? { ...t, status: 'running' } : t));
      const delay = test.type === 'e2e' ? 600 + Math.random() * 800 : 100 + Math.random() * 300;
      await new Promise(r => setTimeout(r, delay));

      const passed = Math.random() > (test.id === 'e4' ? 0.3 : 0.08);
      setTests(prev => prev.map(t => t.id === test.id ? {
        ...t,
        status: passed ? 'passed' : 'failed',
        duration: Math.round(delay),
        error: passed ? undefined : 'Expected "success" but got "error"',
      } : t));
    }
    setIsRunning(false);
  }, []);

  const displayed = tests.filter(t => {
    if (filter !== 'all' && t.type !== filter) return false;
    if (statusFilter !== 'all' && t.status !== statusFilter) return false;
    return true;
  });

  const stats = {
    total: tests.length,
    passed: tests.filter(t => t.status === 'passed').length,
    failed: tests.filter(t => t.status === 'failed').length,
    running: tests.filter(t => t.status === 'running').length,
    pending: tests.filter(t => t.status === 'pending').length,
  };

  const suites = [...new Set(displayed.map(t => t.suite))];

  return (
    <div style={{ background: BG, minHeight: '100vh', padding: 20, fontFamily: 'system-ui, sans-serif', color: TEXT }}>
      <div style={{ maxWidth: 700, margin: '0 auto' }}>
        <h1 style={{ color: QWIK, margin: '0 0 4px' }}>🧪 Тестирование Qwik</h1>
        <p style={{ color: MUTED, margin: '0 0 16px', fontSize: 13 }}>Vitest (unit/component) + Playwright (E2E)</p>

        {/* Stats bar */}
        <div style={{ display: 'flex', gap: 8, marginBottom: 16, flexWrap: 'wrap' }}>
          {[
            { label: 'Всего', value: stats.total, color: TEXT },
            { label: 'Прошли', value: stats.passed, color: GREEN },
            { label: 'Упали', value: stats.failed, color: RED },
            { label: 'В процессе', value: stats.running, color: ORANGE },
            { label: 'Ожидают', value: stats.pending, color: MUTED },
          ].map(s => (
            <div key={s.label} style={{ background: CARD, borderRadius: 8, padding: '8px 14px', border: '1px solid #334155', flex: 1, minWidth: 80, textAlign: 'center' }}>
              <div style={{ color: s.color, fontSize: 22, fontWeight: 700, fontFamily: 'monospace' }}>{s.value}</div>
              <div style={{ color: MUTED, fontSize: 11 }}>{s.label}</div>
            </div>
          ))}
        </div>

        {/* Controls */}
        <div style={{ display: 'flex', gap: 8, marginBottom: 16, flexWrap: 'wrap' }}>
          <button onClick={() => runTests()} disabled={isRunning} style={{ background: isRunning ? '#334155' : QWIK, color: isRunning ? MUTED : '#0f172a', border: 'none', borderRadius: 8, padding: '8px 18px', cursor: isRunning ? 'not-allowed' : 'pointer', fontWeight: 700, fontSize: 13 }}>
            {isRunning ? '⟳ Запуск...' : '▶ Все тесты'}
          </button>
          {(['unit', 'component', 'e2e'] as const).map(t => (
            <button key={t} onClick={() => runTests(t)} disabled={isRunning} style={{ background: CARD, color: typeColors[t], border: \`1px solid \${typeColors[t]}44\`, borderRadius: 8, padding: '8px 14px', cursor: isRunning ? 'not-allowed' : 'pointer', fontSize: 13, fontWeight: 600 }}>
              {typeLabels[t]}
            </button>
          ))}
        </div>

        {/* Filter */}
        <div style={{ display: 'flex', gap: 6, marginBottom: 16, flexWrap: 'wrap' }}>
          {(['all', 'unit', 'component', 'e2e'] as const).map(f => (
            <button key={f} onClick={() => setFilter(f)} style={{ background: filter === f ? (f === 'all' ? QWIK : typeColors[f]) : CARD, color: filter === f ? '#0f172a' : TEXT, border: '1px solid #334155', borderRadius: 6, padding: '4px 12px', cursor: 'pointer', fontSize: 12 }}>
              {f === 'all' ? 'Все' : typeLabels[f]}
            </button>
          ))}
          <div style={{ width: 1, background: '#334155', margin: '0 4px' }} />
          {(['passed', 'failed', 'pending'] as const).map(s => (
            <button key={s} onClick={() => setStatusFilter(statusFilter === s ? 'all' : s)} style={{ background: statusFilter === s ? \`\${statusColors[s]}22\` : CARD, color: statusColors[s], border: \`1px solid \${statusFilter === s ? statusColors[s] : '#334155'}\`, borderRadius: 6, padding: '4px 12px', cursor: 'pointer', fontSize: 12 }}>
              {statusIcons[s]} {s}
            </button>
          ))}
        </div>

        {/* Test list by suite */}
        {suites.map(suite => (
          <div key={suite} style={{ background: CARD, borderRadius: 12, padding: 16, marginBottom: 8, border: '1px solid #334155' }}>
            <div style={{ color: TEXT, fontWeight: 700, fontSize: 14, marginBottom: 8, display: 'flex', justifyContent: 'space-between' }}>
              <span>{suite}</span>
              <span style={{ color: MUTED, fontSize: 12, fontFamily: 'monospace' }}>
                {displayed.filter(t => t.suite === suite && t.status === 'passed').length}/{displayed.filter(t => t.suite === suite).length}
              </span>
            </div>
            {displayed.filter(t => t.suite === suite).map(test => (
              <div key={test.id} style={{ display: 'flex', alignItems: 'center', gap: 10, padding: '5px 0', borderBottom: '1px solid #1e293b' }}>
                <span style={{ color: statusColors[test.status], fontSize: 14, minWidth: 16 }}>{statusIcons[test.status]}</span>
                <span style={{ flex: 1, color: test.status === 'failed' ? RED : test.status === 'passed' ? TEXT : MUTED, fontSize: 13 }}>{test.name}</span>
                <span style={{ background: \`\${typeColors[test.type]}22\`, color: typeColors[test.type], fontSize: 10, padding: '1px 6px', borderRadius: 3, fontWeight: 700 }}>{typeLabels[test.type]}</span>
                {test.duration && <span style={{ color: test.duration > 500 ? ORANGE : MUTED, fontSize: 11, fontFamily: 'monospace', minWidth: 50, textAlign: 'right' }}>{test.duration}мс</span>}
                {test.error && <span style={{ color: RED, fontSize: 10 }}>⚠</span>}
              </div>
            ))}
          </div>
        ))}

        {stats.passed > 0 && !isRunning && (
          <div style={{ background: \`\${GREEN}11\`, borderRadius: 12, padding: 16, border: \`1px solid \${GREEN}33\`, textAlign: 'center' }}>
            <div style={{ color: GREEN, fontWeight: 700, fontSize: 16 }}>
              {stats.failed === 0 ? \`✅ Все \${stats.passed} тестов прошли!\` : \`⚠️ \${stats.passed} прошли, \${stats.failed} упали\`}
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
` }} />
