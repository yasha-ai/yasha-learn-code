# Real-World Dashboard: Всё вместе

## Полный стек TanStack в одном приложении

В этом финальном уроке объединим все изученные инструменты в работающий дашборд:

- **TanStack Query** — загрузка данных, кэширование, мутации
- **TanStack Table** — таблица с сортировкой, фильтрацией и пагинацией
- **TanStack Router концепции** — навигация между разделами
- **TanStack Form концепции** — форма редактирования с валидацией

## Архитектура дашборда

```
Dashboard
├── QueryClientProvider (TanStack Query)
├── Sidebar Navigation
└── Main Content
    ├── Users Table
    │   ├── useQuery(['users']) — загрузка данных
    │   ├── useReactTable — сортировка, фильтрация, пагинация
    │   └── useMutation — удаление пользователя
    ├── User Detail Panel
    │   ├── useQuery(['user', id]) — детали пользователя
    │   └── Edit Form с валидацией
    └── Cache Status Panel
        └── Отображение состояния кэша Query
```

## Ключевые паттерны

### 1. Единый QueryClient на приложение
```typescript
const queryClient = new QueryClient({
  defaultOptions: { queries: { staleTime: 1000 * 60 * 2 } },
})
```

### 2. Инвалидация после мутаций
```typescript
const deleteUser = useMutation({
  mutationFn: (id) => api.deleteUser(id),
  onSuccess: () => queryClient.invalidateQueries({ queryKey: ['users'] }),
})
```

### 3. Optimistic UI + Rollback
```typescript
const updateUser = useMutation({
  mutationFn: api.updateUser,
  onMutate: async (updatedUser) => {
    const prev = queryClient.getQueryData(['users'])
    queryClient.setQueryData(['users'], (old) => old.map(u => u.id === updatedUser.id ? updatedUser : u))
    return { prev }
  },
  onError: (_, __, ctx) => queryClient.setQueryData(['users'], ctx.prev),
  onSettled: () => queryClient.invalidateQueries({ queryKey: ['users'] }),
})
```

// ====== СИМУЛЯЦИЯ QUERY ======
function useSimulatedQuery<T>(fetcher: () => Promise<T>, key: string, enabled = true) {
  const [data, setData] = useState<T | undefined>()
  const [status, setStatus] = useState<QueryStatus>('idle')
  const [fetchCount, setFetchCount] = useState(0)

  useEffect(() => {
    if (!enabled) return
    setStatus('loading')
    fetcher().then(d => {
      setData(d)
      setStatus('success')
    }).catch(() => setStatus('error'))
  }, [key, enabled, fetchCount])

  const refetch = () => setFetchCount(c => c + 1)
  const invalidate = () => setFetchCount(c => c + 1)

  return { data, status, isLoading: status === 'loading', isSuccess: status === 'success', refetch, invalidate }
}

function useSimulatedMutation<T, V>(
  mutFn: (vars: V) => Promise<T>,
  onSuccess?: (data: T, vars: V) => void
) {
  const [status, setStatus] = useState<MutationStatus>('idle')
  const mutate = async (vars: V) => {
    setStatus('pending')
    try {
      const result = await mutFn(vars)
      setStatus('success')
      onSuccess?.(result, vars)
    } catch {
      setStatus('error')
    }
  }
  return { mutate, isPending: status === 'pending', status }
}

function EditUserForm({ user, onSave, onCancel, isSaving }: {
  user: User
  onSave: (values: EditFormValues) => void
  onCancel: () => void
  isSaving: boolean
}) {
  const [values, setValues] = useState<EditFormValues>({
    name: user.name, email: user.email, role: user.role, salary: String(user.salary),
  })
  const [errors, setErrors] = useState<Partial<EditFormValues>>({})

  const validate = () => {
    const errs: Partial<EditFormValues> = {}
    if (!values.name || values.name.length < 2) errs.name = 'Мин. 2 символа'
    if (!values.email || !/^[^@]+@[^@]+\.[^@]+$/.test(values.email)) errs.email = 'Невалидный email'
    if (!values.role) errs.role = 'Обязательно'
    if (!values.salary || isNaN(Number(values.salary))) errs.salary = 'Число'
    setErrors(errs)
    return Object.keys(errs).length === 0
  }

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    if (validate()) onSave(values)
  }

  const inputStyle = (err?: string) => ({
    width: '100%', background: '#0f172a',
    border: \`1px solid \${err ? '#ef4444' : '#334155'}\`,
    borderRadius: '6px', padding: '6px 10px', color: '#e2e8f0', fontSize: '12px', boxSizing: 'border-box' as const,
  })

  return (
    <form onSubmit={handleSubmit}>

        { key: 'email', label: 'Email' },
        { key: 'role', label: 'Должность' },

  )
}
