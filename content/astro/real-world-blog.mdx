# Полноценный блог на Astro

Этот урок — финальный проект: мы разберём архитектуру настоящего блога с content collections, RSS-лентой, тёмной темой и поиском.

## Архитектура проекта

```
src/
├── content/
│   ├── config.ts          ← Схемы коллекций
│   └── blog/
│       ├── first-post.md
│       └── second-post.mdx
├── layouts/
│   ├── BaseLayout.astro   ← HTML-обёртка
│   ├── BlogLayout.astro   ← Шапка + навигация
│   └── PostLayout.astro   ← Разметка поста
├── pages/
│   ├── index.astro        ← Главная
│   ├── blog/
│   │   ├── index.astro    ← Список постов
│   │   └── [slug].astro   ← Страница поста
│   └── rss.xml.ts         ← RSS-лента
└── components/
    ├── PostCard.astro
    ├── TagCloud.astro
    └── SearchBar.astro    ← (React, client:load)
```

## Content Collections — схема

```ts
// src/content/config.ts
import { defineCollection, z } from 'astro:content';

const blog = defineCollection({
  type: 'content',
  schema: z.object({
    title: z.string(),
    description: z.string(),
    pubDate: z.date(),
    updatedDate: z.date().optional(),
    heroImage: z.string().optional(),
    tags: z.array(z.string()).default([]),
    author: z.string().default('Редакция'),
    draft: z.boolean().default(false),
    readingTime: z.number().optional(),
  }),
});

export const collections = { blog };
```

## Получение и сортировка постов

```ts
// src/pages/blog/index.astro
---
import { getCollection } from 'astro:content';

const allPosts = await getCollection('blog', ({ data }) => {
  return import.meta.env.PROD ? !data.draft : true;
});

const posts = allPosts.sort((a, b) =>
  b.data.pubDate.valueOf() - a.data.pubDate.valueOf()
);

const tags = [...new Set(posts.flatMap(p => p.data.tags))];
---
```

## RSS-лента

```ts
// src/pages/rss.xml.ts
import rss from '@astrojs/rss';
import { getCollection } from 'astro:content';

export async function GET(context) {
  const posts = await getCollection('blog');
  return rss({
    title: 'Мой Astro блог',
    description: 'Статьи о веб-разработке',
    site: context.site,
    items: posts.map(post => ({
      title: post.data.title,
      pubDate: post.data.pubDate,
      description: post.data.description,
      link: '/blog/' + post.slug,
    })),
  });
}
```

function useTheme() {
  const [dark, setDark] = useState(true);
  const bg =     dark ? C.astroBg     : C.lightBg;
  const card =   dark ? C.astroCard   : C.lightCard;
  const border = dark ? C.astroBorder : C.lightBorder;
  const text =   dark ? C.astroText   : C.lightText;
  const muted =  dark ? C.astroMuted  : C.lightMuted;
  return { dark, setDark, bg, card, border, text, muted };
}

function TagBadge({ tag, active, onClick, color, dark }) {
  const border = dark ? C.astroBorder : C.lightBorder;
  const muted = dark ? C.astroMuted : C.lightMuted;
  return (
    <button onClick={() => onClick(tag)} style={{
      padding: '3px 10px', borderRadius: 12, border: '1px solid',
      borderColor: active ? (color || C.astroOrange) : border,
      background: active ? (color || C.astroOrange) + '25' : 'transparent',
      color: active ? (color || C.astroOrange) : muted,
      cursor: 'pointer', fontSize: 11, transition: 'all 0.2s'
    }}>#{tag}</button>
  );
}

function ReadingProgress({ dark }) {
  const [progress, setProgress] = useState(0);
  useEffect(() => {
    let p = 0;
    const iv = setInterval(() => {
      p += 0.8;
      if (p > 100) p = 100;
      setProgress(p);
    }, 80);
    return () => clearInterval(iv);
  }, []);
  return (
    <div style={{ position: 'sticky', top: 0, left: 0, right: 0, height: 3, background: dark ? C.astroBorder : C.lightBorder, zIndex: 100 }}>
      <div style={{ height: '100%', width: progress + '%', background: C.astroOrange, transition: 'width 0.1s linear', borderRadius: '0 2px 2px 0' }} />
    </div>
  );
}

function PostDetail({ post, onBack, th }) {
  const prevIdx = POSTS.findIndex(p => p.id === post.id) + 1;

  const next = nextIdx >= 0 ? POSTS[nextIdx] : null;

  return (
    <div>
      <ReadingProgress dark={th.dark} />
      <div style={{ padding: '20px 20px 0' }}>
        <button onClick={onBack} style={{
          display: 'flex', alignItems: 'center', gap: 6,
