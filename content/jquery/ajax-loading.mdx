# Загрузка контента без перезагрузки

Динамическая подгрузка контента на страницу.

## Infinite scroll

Бесконечная прокрутка

```javascript
// Пример кода для Infinite scroll
$('.example').text('Hello from Infinite scroll');
```

## Load more кнопка

Подгрузка по клику

```javascript
// Пример кода для Load more кнопка
$('.example').text('Hello from Load more кнопка');
```

## Индикаторы загрузки

Спиннеры и прогресс

```javascript
// Пример кода для Индикаторы загрузки
$('.example').text('Hello from Индикаторы загрузки');
```

## 🚀 Практика

## Интерактивный пример

<Playground
  template="static"
  files={{
    "/index.html": `<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
  <style>
    body { background: #1e1e2e; color: #cdd6f4; font-family: sans-serif; padding: 20px; margin: 0; }
    button { background: #89b4fa; color: #1e1e2e; border: none; padding: 8px 16px; border-radius: 6px; cursor: pointer; margin: 4px; font-size: 14px; }
    button:hover { background: #74c7ec; }
    button:disabled { background: #45475a; color: #585b70; cursor: not-allowed; }
    .box { background: #313244; border-radius: 8px; padding: 16px; margin: 10px 0; }
    .spinner { display: none; border: 3px solid #45475a; border-top: 3px solid #89b4fa; border-radius: 50%; width: 28px; height: 28px; animation: spin 0.8s linear infinite; margin: 10px auto; }
    @keyframes spin { to { transform: rotate(360deg); } }
    .progress-bar { height: 8px; background: #45475a; border-radius: 4px; overflow: hidden; margin: 10px 0; }
    .progress-fill { height: 100%; width: 0; background: #89b4fa; border-radius: 4px; transition: width 0.3s; }
    .item { background: #45475a; border-radius: 6px; padding: 8px 12px; margin: 4px 0; display: none; }
    #status { font-size: 13px; color: #f9e2af; min-height: 20px; }
  </style>
</head>
<body>
  <h3 style="color:#cba6f7">Индикаторы загрузки</h3>
  <div class="box">
    <button id="loadSimple">Загрузить (спиннер)</button>
    <button id="loadProgress">Загрузить (прогресс-бар)</button>
    <button id="loadMore">Load More</button>
  </div>
  <div class="box">
    <div id="status"></div>
    <div class="spinner" id="spinner"></div>
    <div class="progress-bar" id="progressBar" style="display:none"><div class="progress-fill" id="progressFill"></div></div>
    <div id="content"></div>
  </div>
  <script>
    var items = ['Элемент 1', 'Элемент 2', 'Элемент 3', 'Элемент 4', 'Элемент 5'];
    var loaded = 0;

    $('#loadSimple').click(function() {
      var $btn = $(this).prop('disabled', true);
      $('#status').text('Загружаю данные...');
      $('#spinner').show();
      $('#content').empty();

      setTimeout(function() {
        $('#spinner').hide();
        $('#status').text('');
        var html = '';
        $.each(items, function(i, v) { html += '<div class="item" style="display:block">' + v + '</div>'; });
        $('#content').html(html).children().hide().each(function(i) {
          $(this).delay(i * 150).fadeIn(300);
        });
        $btn.prop('disabled', false);
      }, 1500);
    });

    $('#loadProgress').click(function() {
      var $btn = $(this).prop('disabled', true);
      var $fill = $('#progressFill');
      $('#progressBar').show();
      $('#content').empty();
      var progress = 0;

      var interval = setInterval(function() {
        progress += 20;
        $fill.css('width', progress + '%');
        $('#status').text('Загрузка: ' + progress + '%');
        if (progress >= 100) {
          clearInterval(interval);
          setTimeout(function() {
            $('#progressBar').hide();
            $fill.css('width', '0');
            $('#status').text('✅ Загружено!');
            $('#content').html('<div class="item" style="display:block;color:#a6e3a1">Данные успешно получены!</div>');
            $btn.prop('disabled', false);
          }, 200);
        }
      }, 300);
    });

    $('#loadMore').click(function() {
      var $btn = $(this);
      $btn.text('Загружаю...').prop('disabled', true);
      setTimeout(function() {
        loaded += 2;
        var html = '';
        for (var i = loaded - 1; i <= loaded; i++) {
          html += '<div class="item" style="display:block">Новый элемент #' + i + '</div>';
        }
        $(html).hide().appendTo('#content').fadeIn(400);
        $btn.text('Load More').prop('disabled', false);
      }, 800);
    });
  </script>
</body>
</html>`,
  }}
/>
