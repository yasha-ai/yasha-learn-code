# Анимации

jQuery включает встроенные методы для создания анимаций.

## Показать/скрыть

.show(), .hide(), .toggle()

```javascript
// Пример кода для Показать/скрыть
$('.example').text('Hello from Показать/скрыть');
```

## Fade эффекты

.fadeIn(), .fadeOut(), .fadeTo(), .fadeToggle()

```javascript
// Пример кода для Fade эффекты
$('.example').text('Hello from Fade эффекты');
```

## Slide эффекты

.slideDown(), .slideUp(), .slideToggle()

```javascript
// Пример кода для Slide эффекты
$('.example').text('Hello from Slide эффекты');
```

## Пользовательские анимации

.animate()

```javascript
// Пример кода для Пользовательские анимации
$('.example').text('Hello from Пользовательские анимации');
```

## Управление очередью

.stop(), .delay(), .finish()

```javascript
// Пример кода для Управление очередью
$('.example').text('Hello from Управление очередью');
```

## Интерактивный пример

<Playground
  template="static"
  files={{
    "/index.html": `<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
  <style>
    body { background: #1e1e2e; color: #cdd6f4; font-family: sans-serif; padding: 20px; margin: 0; }
    button { background: #89b4fa; color: #1e1e2e; border: none; padding: 7px 13px; border-radius: 6px; cursor: pointer; margin: 3px; font-size: 13px; }
    button:hover { background: #74c7ec; }
    .box { background: #313244; border-radius: 8px; padding: 16px; margin: 10px 0; }
    #target { background: #89b4fa; color: #1e1e2e; border-radius: 8px; padding: 20px; text-align: center; font-weight: bold; font-size: 16px; margin: 10px 0; width: 200px; }
    .section-title { font-size: 12px; color: #bac2de; margin: 8px 0 4px; text-transform: uppercase; letter-spacing: 1px; }
  </style>
</head>
<body>
  <h3 style="color:#cba6f7">Анимации jQuery</h3>
  <div class="box">
    <div id="target">Анимируй меня!</div>
    <div class="section-title">Показать / скрыть</div>
    <button id="btnShow">.show()</button>
    <button id="btnHide">.hide()</button>
    <button id="btnToggle">.toggle()</button>
    <div class="section-title">Fade</div>
    <button id="btnFadeIn">.fadeIn()</button>
    <button id="btnFadeOut">.fadeOut()</button>
    <button id="btnFadeToggle">.fadeToggle()</button>
    <button id="btnFadeTo">.fadeTo(0.3)</button>
    <div class="section-title">Slide</div>
    <button id="btnSlideDown">.slideDown()</button>
    <button id="btnSlideUp">.slideUp()</button>
    <button id="btnSlideToggle">.slideToggle()</button>
    <div class="section-title">Animate</div>
    <button id="btnAnimWidth">.animate() ширина</button>
    <button id="btnAnimMove">.animate() отступ</button>
    <button id="btnStop">.stop()</button>
    <button id="btnReset">Сбросить</button>
  </div>
  <script>
    var $t = $('#target');
    $('#btnShow').click(function()        { $t.show('slow'); });
    $('#btnHide').click(function()        { $t.hide('slow'); });
    $('#btnToggle').click(function()      { $t.toggle('slow'); });
    $('#btnFadeIn').click(function()      { $t.fadeIn(600); });
    $('#btnFadeOut').click(function()     { $t.fadeOut(600); });
    $('#btnFadeToggle').click(function()  { $t.fadeToggle(600); });
    $('#btnFadeTo').click(function()      { $t.fadeTo(600, 0.3); });
    $('#btnSlideDown').click(function()   { $t.slideDown(600); });
    $('#btnSlideUp').click(function()     { $t.slideUp(600); });
    $('#btnSlideToggle').click(function() { $t.slideToggle(600); });
    $('#btnAnimWidth').click(function()   { $t.animate({ width: $t.width() === 200 ? 320 : 200 }, 600); });
    $('#btnAnimMove').click(function()    { $t.animate({ marginLeft: parseInt($t.css('marginLeft')) === 0 ? 80 : 0 }, 600); });
    $('#btnStop').click(function()        { $t.stop(true, false); });
    $('#btnReset').click(function()       { $t.stop(true).show().css({ opacity: 1, width: 200, marginLeft: 0 }); });
  </script>
</body>
</html>`,
  }}
/>
