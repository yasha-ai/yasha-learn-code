# $.get() и $.post()

Упрощенные методы для GET и POST запросов.

## $.get()

GET запросы

```javascript
// Пример кода для $.get()
$('.example').text('Hello from $.get()');
```

## $.post()

POST запросы

```javascript
// Пример кода для $.post()
$('.example').text('Hello from $.post()');
```

## $.getScript()

Загрузка JS файлов

```javascript
// Пример кода для $.getScript()
$('.example').text('Hello from $.getScript()');
```

## $.load()

Загрузка HTML в элемент

```javascript
// Пример кода для $.load()
$('.example').text('Hello from $.load()');
```

    <button id="doGet">$.get()</button>
    <button id="doPost">$.post()</button>
    <button id="doGetJSON">$.getJSON()</button>
    <button id="doLoad">$.load()</button>
  </div>

    <div id="output" class="output">Выберите метод...</div>
  </div>

  <script>
    function show(method, loading) {
      $('#output').html('<span class="loading">⏳ ' + method + ' — отправляю запрос...</span>');
    }
    function result(method, data) {
      $('#output').html('<span class="method">' + method + '</span> ответ:\\n' + JSON.stringify(data, null, 2));
    }

    // $.get(url, callback)
    $('#doGet').click(function() {
      show('$.get()');
      setTimeout(function() {
        // $.get('/api/data', function(data) { ... })
        result('$.get()', { items: ['foo', 'bar', 'baz'], total: 3 });
      }, 700);
    });

    // $.post(url, data, callback)
    $('#doPost').click(function() {
      show('$.post()');
      var sent = { username: 'user42', message: 'Привет!' };
      setTimeout(function() {
        // $.post('/api/message', sent, function(data) { ... })
        result('$.post()', { success: true, echo: sent, id: 99 });
      }, 700);
    });

    // $.getJSON(url, callback)
    $('#doGetJSON').click(function() {
      show('$.getJSON()');
      setTimeout(function() {
        // $.getJSON('/api/config', function(json) { ... })
        var json = { theme: 'dark', lang: 'ru', version: '2.1.0' };
        result('$.getJSON()', json);
      }, 700);
    });

    // $('selector').load(url)
    $('#doLoad').click(function() {
      show('$.load()');
      setTimeout(function() {
        // $('#container').load('/partial.html')
        $('#output').html('<span class="method">$().load()</span> — вставляет HTML фрагмент в элемент:\\n\\n' +
          '&lt;div id="container"&gt;\\n  &lt;!-- HTML с сервера вставляется сюда --&gt;\\n' +
          '  &lt;p&gt;Загруженный контент&lt;/p&gt;\\n&lt;/div&gt;');
      }, 700);
    });
  </script>
</body>
