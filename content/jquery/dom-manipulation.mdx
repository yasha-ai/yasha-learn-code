# Манипуляция DOM

jQuery предоставляет мощные методы для изменения структуры страницы.

## Добавление элементов

.append(), .prepend(), .after(), .before()

```javascript
// Пример кода для Добавление элементов
$('.example').text('Hello from Добавление элементов');
```

## Удаление элементов

.remove(), .empty(), .detach()

```javascript
// Пример кода для Удаление элементов
$('.example').text('Hello from Удаление элементов');
```

## Замена элементов

.replaceWith(), .replaceAll()

```javascript
// Пример кода для Замена элементов
$('.example').text('Hello from Замена элементов');
```

## Клонирование

.clone()

```javascript
// Пример кода для Клонирование
$('.example').text('Hello from Клонирование');
```

## Оборачивание

.wrap(), .wrapAll(), .unwrap()

```javascript
// Пример кода для Оборачивание
$('.example').text('Hello from Оборачивание');
```

  template="static"
  files={{
    "/index.html": `<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
  <style>
    body { background: #1e1e2e; color: #cdd6f4; font-family: sans-serif; padding: 20px; margin: 0; }
    button { background: #89b4fa; color: #1e1e2e; border: none; padding: 7px 12px; border-radius: 6px; cursor: pointer; margin: 2px; font-size: 13px; }
    button:hover { background: #74c7ec; }
    button.danger { background: #f38ba8; }
    input[type=text] { background: #313244; color: #cdd6f4; border: 1px solid #45475a; padding: 6px 10px; border-radius: 4px; width: 140px; }
    .box { background: #313244; border-radius: 8px; padding: 16px; margin: 10px 0; }
    .item { background: #45475a; border-radius: 6px; padding: 8px 14px; margin: 4px 0; display: flex; justify-content: space-between; align-items: center; }
    .item .del { background: #f38ba8; color: #1e1e2e; border: none; border-radius: 4px; padding: 2px 8px; cursor: pointer; font-size: 12px; }
    .item .wrap-btn { background: #cba6f7; color: #1e1e2e; border: none; border-radius: 4px; padding: 2px 8px; cursor: pointer; font-size: 12px; margin-left: 4px; }
    .wrapper { border: 2px dashed #cba6f7; border-radius: 6px; padding: 4px; margin: 4px 0; }
    .clone-badge { background: #a6e3a1; color: #1e1e2e; border-radius: 3px; padding: 1px 5px; font-size: 11px; margin-left: 6px; }
    #log { background: #181825; border-radius: 6px; padding: 10px; font-family: monospace; font-size: 12px; color: #a6e3a1; min-height: 24px; }
  </style>
</head>
<body>
  <h3 style="color:#cba6f7">Манипуляция DOM</h3>
  <div class="box">
    <input id="newText" type="text" placeholder="Текст элемента">
    <button id="btnAppend">.append()</button>
    <button id="btnPrepend">.prepend()</button>
    <button id="btnAfter">.after()</button>
    <button id="btnBefore">.before()</button>
    <button id="btnEmpty" class="danger">.empty()</button>
    <button id="btnClone">.clone()</button>
    <button id="btnWrapAll">.wrapAll()</button>
    <button id="btnUnwrap">.unwrap()</button>
    <button id="btnHtml">.html()</button>
  </div>
  <div id="list" class="box" style="min-height:60px"></div>
  <div class="box"><strong style="font-size:12px;color:#bac2de">Лог:</strong> <span id="log">—</span></div>
  <script>
    var counter = 1;
    function log(msg) { $('#log').text(msg); }
    function makeItem(text, extra) {
      return $('<div class="item">' +
        '<span>' + text + (extra ? '<span class="clone-badge">' + extra + '</span>' : '') + '</span>' +
        '<span>' +
        '<button class="wrap-btn">wrap</button>' +
        '<button class="del">✕</button>' +
        '</span></div>');
    }

    // Initial items
    $('#list').append(makeItem('Элемент A')).append(makeItem('Элемент B'));

    function getText() { return $('#newText').val().trim() || ('Элемент ' + counter++); }

    $('#btnAppend').click(function()  { var t = getText(); $('#list').append(makeItem(t));   log(".append() — добавлен в конец: " + t); });
    $('#btnPrepend').click(function() { var t = getText(); $('#list').prepend(makeItem(t));  log(".prepend() — добавлен в начало: " + t); });
    $('#btnAfter').click(function()   { var t = getText(); $('#list').after(makeItem(t));    log(".after() — добавлен после #list: " + t); });
    $('#btnBefore').click(function()  { var t = getText(); $('#list').before(makeItem(t));   log(".before() — добавлен перед #list: " + t); });
    $('#btnEmpty').click(function()   { $('#list').empty();                                  log(".empty() — дочерние элементы удалены"); });
    $('#btnClone').click(function()   {
      var $first = $('#list .item:first');
      if ($first.length) { $first.clone().appendTo('#list').find('span:first').append('<span class="clone-badge">clone</span>'); log(".clone() — первый элемент клонирован"); }
      else { log("Нет элементов для клонирования"); }
    });
    $('#btnWrapAll').click(function() {
      var $items = $('#list .item:not(.wrapped)').addClass('wrapped');
      if ($items.length) { $items.wrapAll('<div class="wrapper"></div>'); log(".wrapAll() — элементы обёрнуты в .wrapper"); }
    });
    $('#btnUnwrap').click(function()  { $('#list .item').unwrap(); log(".unwrap() — обёртка удалена"); });
    $('#btnHtml').click(function()    {
      var current = $('#list').html();
      log(".html() длина: " + current.length + " симв. | .text() = " + $('#list').text().substring(0, 40) + "...");
    });
    $(document).on('click', '.del', function() {
      var text = $(this).closest('.item').find('span:first').text().trim().substring(0, 20);
      $(this).closest('.item').remove();
      log(".remove() — удалён: " + text);
    });
    $(document).on('click', '.wrap-btn', function() {
      var $item = $(this).closest('.item');
      if (!$item.parent().hasClass('wrapper')) {
        $item.wrap('<div class="wrapper"></div>');
        log(".wrap() — элемент обёрнут");
      }
    });
  </script>
</body>
</html>`,
  }}
/>
