### –†–∞—Å—à–∏—Ä–µ–Ω–∏–µ –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–æ–≤ —Å Generics

–†–∞—Å—à–∏—Ä–µ–Ω–∏–µ –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–æ–≤ —Å—Ç–∞–Ω–æ–≤–∏—Ç—Å—è –µ—â–µ –º–æ—â–Ω–µ–µ, –∫–æ–≥–¥–∞ –º—ã –∫–æ–º–±–∏–Ω–∏—Ä—É–µ–º –µ–≥–æ —Å –æ–±–æ–±—â–µ–Ω–∏—è–º–∏ (generics). –≠—Ç–æ –ø–æ–∑–≤–æ–ª—è–µ—Ç –Ω–∞–º —Å–æ–∑–¥–∞–≤–∞—Ç—å –ø–æ–≤—Ç–æ—Ä–Ω–æ –∏—Å–ø–æ–ª—å–∑—É–µ–º—ã–µ —à–∞–±–ª–æ–Ω—ã —Ç–∏–ø–æ–≤.

```typescript
interface CrudRepository<T, ID = string> {
  create(item: T): T;
  findById(id: ID): T | undefined;
  update(id: ID, item: Partial<T>): T | undefined;
  delete(id: ID): boolean;
  findAll(): T[];

// –†–∞—Å—à–∏—Ä—è–µ–º CrudRepository –¥–ª—è –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–≥–æ —Ç–∏–ø–∞ —Å—É—â–Ω–æ—Å—Ç–∏ - User

  // –ü—Ä–∏–º–µ–Ω–∏–º UserProfile –∏–∑ –ø—Ä–∏–º–µ—Ä–∞ –≤—ã—à–µ
}

// –ü—Ä–∏–º–µ—Ä —Ä–µ–∞–ª–∏–∑–∞—Ü–∏–∏ (–¥–ª—è –¥–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏–∏)
class InMemoryUserRepository implements UserRepository {
  private users: Map<string, UserProfile> = new new Map();

  create(user: UserProfile): UserProfile {
    this.users.set(user.id, user);
    return user;
  }
  findById(id: string): UserProfile | undefined {
    return this.users.get(id);
  }
  update(id: string, updates: Partial<UserProfile>): UserProfile | undefined {
    const user = this.users.get(id);
    if (!user) return undefined;
    const updatedUser = { ...user, ...updates };
    this.users.set(id, updatedUser);
    return updatedUser;
  }
  delete(id: string): boolean {
    return this.users.delete(id);
  }
  findAll(): UserProfile[] {
    return Array.from(this.users.values());
  }
  findByEmail(email: string): UserProfile | undefined {
    return Array.from(this.users.values()).find((u) => u.email === email);
  }
}

const userRepository = new InMemoryUserRepository();

const alice = userRepository.findById("1");
console.log(alice);
```
