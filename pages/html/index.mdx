## Что освоишь?

После прохождения курса ты сможешь:
- Создавать семантичную и доступную разметку
- Работать с формами, медиа и интерактивными элементами
- Рисовать и анимировать с Canvas/SVG
- Создавать Web Components
- Оптимизировать SEO и доступность
- Использовать современные HTML5 API

<Sandpack
  template="vanilla"
  files={{
    "/index.html": `<!DOCTYPE html>
<html lang="ru">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>HTML Скелет</title>
  <link rel="stylesheet" href="/styles.css">
</head>
<body>
  <header>
    <h1>Привет, HTML!</h1>
    <nav>
      <button id="toggle-button">Изменить тему</button>
    </nav>
  </header>

  <main>
    <section id="content">
      <h2>Основной контент</h2>
      <p>Это пример HTML-скелета. Нажмите кнопку, чтобы изменить тему.</p>
      <input type="text" id="name-input" placeholder="Введите ваше имя">
      <button id="greet-button">Приветствовать!</button>
      <p id="greeting-message"></p>
    </section>
  </main>

  <footer>
    <p>&copy; 2024 Мой сайт</p>
  </footer>

  <script src="/index.js"></script>
</body>
</html>`,
    "/styles.css": `body {
  font-family: sans-serif;
  margin: 0;

    "/index.js": `document.addEventListener('DOMContentLoaded', () => {
  const toggleButton = document.getElementById('toggle-button');
  const body = document.body;
  const nameInput = document.getElementById('name-input');
  const greetButton = document.getElementById('greet-button');
  const greetingMessage = document.getElementById('greeting-message');

  toggleButton.addEventListener('click', () => {
    body.classList.toggle('light-theme');
  });

  greetButton.addEventListener('click', () => {
    const name = nameInput.value;
    if (name) {
      greetingMessage.textContent = \`Привет, \${name}!\`;
    } else {
      greetingMessage.textContent = 'Введите имя!';
    }
  });
});
`
