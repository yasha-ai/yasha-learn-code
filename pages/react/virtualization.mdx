### –û—Å–Ω–æ–≤–Ω–∞—è —Ç–µ–æ—Ä–∏—è —Å –ø—Ä–∏–º–µ—Ä–∞–º–∏: –§–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω–∞—è –≤—ã—Å–æ—Ç–∞ —ç–ª–µ–º–µ–Ω—Ç–æ–≤

–ù–∞—á–Ω–µ–º —Å —Å–∞–º–æ–≥–æ –ø—Ä–æ—Å—Ç–æ–≥–æ, –Ω–æ –º–æ—â–Ω–æ–≥–æ —Å–ª—É—á–∞—è: –≤—Å–µ —ç–ª–µ–º–µ–Ω—Ç—ã —Å–ø–∏—Å–∫–∞ –∏–º–µ—é—Ç –æ–¥–∏–Ω–∞–∫–æ–≤—É—é –≤—ã—Å–æ—Ç—É.

#### ### –ò–Ω—Ç–µ—Ä—Ñ–µ–π—Å—ã –∏ —Ç–∏–ø—ã –¥–ª—è –Ω–∞—à–µ–≥–æ –≤–∏—Ä—Ç—É–∞–ª—å–Ω–æ–≥–æ —Å–ø–∏—Å–∫–∞

```typescript
// –¢–∏–ø –¥–∞–Ω–Ω—ã—Ö –¥–ª—è —ç–ª–µ–º–µ–Ω—Ç–∞ —Å–ø–∏—Å–∫–∞

  // –ú–æ–≥—É—Ç –±—ã—Ç—å –ª—é–±—ã–µ –¥—Ä—É–≥–∏–µ –¥–∞–Ω–Ω—ã–µ
}

// –ü—Ä–æ–ø—Å—ã –¥–ª—è –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–∞ –≤–∏—Ä—Ç—É–∞–ª—å–Ω–æ–≥–æ —Å–ø–∏—Å–∫–∞
interface VirtualListProps<T extends ListItem> {
  data: T[]; // –ú–∞—Å—Å–∏–≤ –¥–∞–Ω–Ω—ã—Ö –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è
  itemHeight: number; // –§–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω–∞—è –≤—ã—Å–æ—Ç–∞ –∫–∞–∂–¥–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞ –≤ –ø–∏–∫—Å–µ–ª—è—Ö
  containerHeight: number; // –í—ã—Å–æ—Ç–∞ –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä–∞, –≤ –∫–æ—Ç–æ—Ä–æ–º –æ—Ç–æ–±—Ä–∞–∂–∞–µ—Ç—Å—è —Å–ø–∏—Å–æ–∫
  renderItem: (item: T, index: number) => JSX.Element; // –§—É–Ω–∫—Ü–∏—è –¥–ª—è —Ä–µ–Ω–¥–µ—Ä–∞ –∫–∞–∂–¥–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
  overscanCount?: number; // –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –¥–ª—è —Ä–µ–Ω–¥–µ—Ä–∞ –∑–∞ –ø—Ä–µ–¥–µ–ª–∞–º–∏ –≤–∏–¥–∏–º–æ–π –æ–±–ª–∞—Å—Ç–∏ (–±—É—Ñ–µ—Ä)
}

// –°–æ—Å—Ç–æ—è–Ω–∏–µ –Ω–∞—à–µ–≥–æ –≤–∏—Ä—Ç—É–∞–ª—å–Ω–æ–≥–æ —Å–ø–∏—Å–∫–∞ (–≤ —Ä–µ–∞–ª—å–Ω–æ–º –ø—Ä–∏–ª–æ–∂–µ–Ω–∏–∏ —ç—Ç–æ –±—ã–ª –±—ã useState –∏–ª–∏ store)
interface VirtualListState {
  scrollTop: number; // –¢–µ–∫—É—â–∞—è –ø–æ–∑–∏—Ü–∏—è –ø—Ä–æ–∫—Ä—É—Ç–∫–∏
  visibleStartIndex: number; // –ò–Ω–¥–µ–∫—Å –ø–µ—Ä–≤–æ–≥–æ –≤–∏–¥–∏–º–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
  visibleEndIndex: number; // –ò–Ω–¥–µ–∫—Å –ø–æ—Å–ª–µ–¥–Ω–µ–≥–æ –≤–∏–¥–∏–º–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
```

#### ### –†–∞—Å—á–µ—Ç –≤–∏–¥–∏–º–æ–π –æ–±–ª–∞—Å—Ç–∏

–°–∞–º–æ–µ –≥–ª–∞–≤–Ω–æ–µ ‚Äî —ç—Ç–æ –æ–ø—Ä–µ–¥–µ–ª–∏—Ç—å, –∫–∞–∫–∏–µ —ç–ª–µ–º–µ–Ω—Ç—ã –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å –≤–∏–¥–Ω—ã. –≠—Ç–æ –∑–∞–≤–∏—Å–∏—Ç –æ—Ç —Ç–µ–∫—É—â–µ–π –ø–æ–∑–∏—Ü–∏–∏ —Å–∫—Ä–æ–ª–ª–∞ (`scrollTop`), –≤—ã—Å–æ—Ç—ã –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä–∞ (`containerHeight`) –∏ –≤—ã—Å–æ—Ç—ã –∫–∞–∂–¥–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞ (`itemHeight`).

```typescript
// –í—Å–ø–æ–º–æ–≥–∞—Ç–µ–ª—å–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –¥–ª—è —Ä–∞—Å—á–µ—Ç–∞ –≤–∏–¥–∏–º–æ–≥–æ –¥–∏–∞–ø–∞–∑–æ–Ω–∞
function calculateVisibleRange<T extends ListItem>(
  props: Pick<VirtualListProps<T>, 'data' | 'itemHeight' | 'containerHeight' | 'overscanCount'>,
  scrollTop: number
): { startIndex: number; endIndex: number } {
  const { data, itemHeight, containerHeight, overscanCount = 3 } = props;
  const totalItems = data.length;

  // –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º, —Å–∫–æ–ª—å–∫–æ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –ø–æ–º–µ—â–∞–µ—Ç—Å—è –≤ –≤–∏–¥–∏–º–æ–π –æ–±–ª–∞—Å—Ç–∏
  const itemsInView = Math.ceil(containerHeight / itemHeight);

  // –û–ø—Ä–µ–¥–µ–ª—è–µ–º –Ω–∞—á–∞–ª—å–Ω—ã–π –∏–Ω–¥–µ–∫—Å –≤–∏–¥–∏–º–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
  let startIndex = Math.floor(scrollTop / itemHeight);

  // –û–ø—Ä–µ–¥–µ–ª—è–µ–º –∫–æ–Ω–µ—á–Ω—ã–π –∏–Ω–¥–µ–∫—Å –≤–∏–¥–∏–º–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
  let endIndex = Math.min(totalItems - 1, startIndex + itemsInView);

  // –î–æ–±–∞–≤–ª—è–µ–º –±—É—Ñ–µ—Ä (overscan) –¥–ª—è –ø–ª–∞–≤–Ω–æ–π –ø—Ä–æ–∫—Ä—É—Ç–∫–∏
  startIndex = Math.max(0, startIndex - overscanCount);
  endIndex = Math.min(totalItems - 1, endIndex + overscanCount);

  return { startIndex, endIndex };
}
```

#### ### –ü—Ä–∏–º–µ—Ä –∫–æ–Ω—Ü–µ–ø—Ç—É–∞–ª—å–Ω–æ–≥–æ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–∞ VirtualList

–î–∞–≤–∞–π—Ç–µ —Å–æ–±–µ—Ä–µ–º —ç—Ç–æ –≤ –∫–æ–Ω—Ü–µ–ø—Ç—É–∞–ª—å–Ω—ã–π –∫–æ–º–ø–æ–Ω–µ–Ω—Ç, –∫–æ—Ç–æ—Ä—ã–π –ø–æ–∫–∞–∑—ã–≤–∞–µ—Ç, –∫–∞–∫ —ç—Ç–æ –º–æ–≥–ª–æ –±—ã –≤—ã–≥–ª—è–¥–µ—Ç—å –≤ React –∏–ª–∏ –¥—Ä—É–≥–æ–º —Ñ—Ä–µ–π–º–≤–æ—Ä–∫–µ. –ú—ã –±—É–¥–µ–º —Ñ–æ–∫—É—Å–∏—Ä–æ–≤–∞—Ç—å—Å—è –Ω–∞ –ª–æ–≥–∏–∫–µ TypeScript.

```typescript
// –ö–æ–Ω—Ü–µ–ø—Ç—É–∞–ª—å–Ω—ã–π –∫–ª–∞—Å—Å, –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—é—â–∏–π –ª–æ–≥–∏–∫—É VirtualList
class VirtualListManager<T extends ListItem> {
  private props: VirtualListProps<T>;
  private state: VirtualListState;
  private totalHeight: number; // –û–±—â–∞—è –≤—ã—Å–æ—Ç–∞ –≤—Å–µ–≥–æ –∫–æ–Ω—Ç–µ–Ω—Ç–∞

  constructor(props: VirtualListProps<T>) {
    this.props = props;
    this.state = {
      scrollTop: 0,
      visibleStartIndex: 0,
      visibleEndIndex: 0,
    };
    this.totalHeight = props.data.length * props.itemHeight;

    // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º –≤–∏–¥–∏–º—ã–π –¥–∏–∞–ø–∞–∑–æ–Ω
    this.updateVisibleRange(0);
  }

  // –û–±–Ω–æ–≤–ª—è–µ—Ç –ø—Ä–æ–ø—Å—ã (–Ω–∞–ø—Ä–∏–º–µ—Ä, –ø—Ä–∏ –∏–∑–º–µ–Ω–µ–Ω–∏–∏ –¥–∞–Ω–Ω—ã—Ö)
  public updateProps(newProps: VirtualListProps<T>): void {
    this.props = newProps;
    this.totalHeight = newProps.data.length * newProps.itemHeight;
    this.updateVisibleRange(this.state.scrollTop);
  }

  // –û–±—Ä–∞–±–æ—Ç—á–∏–∫ —Å–æ–±—ã—Ç–∏—è –ø—Ä–æ–∫—Ä—É—Ç–∫–∏
  public handleScroll(newScrollTop: number): void {
    if (newScrollTop === this.state.scrollTop) {
      return; // –û–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è: –µ—Å–ª–∏ —Å–∫—Ä–æ–ª–ª –Ω–µ –∏–∑–º–µ–Ω–∏–ª—Å—è, –Ω–∏—á–µ–≥–æ –Ω–µ –¥–µ–ª–∞–µ–º
    }
    this.state.scrollTop = newScrollTop;
    this.updateVisibleRange(newScrollTop);
  }

  // –ú–µ—Ç–æ–¥ –¥–ª—è –ø–µ—Ä–µ—Å—á–µ—Ç–∞ –≤–∏–¥–∏–º—ã—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –∏ –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è —Å–æ—Å—Ç–æ—è–Ω–∏—è
  private updateVisibleRange(scrollTop: number): void {
    const { startIndex, endIndex } = calculateVisibleRange(this.props, scrollTop);
    if (startIndex !== this.state.visibleStartIndex || endIndex !== this.state.visibleEndIndex) {
      this.state.visibleStartIndex = startIndex;
      this.state.visibleEndIndex = endIndex;
      // –í —Ä–µ–∞–ª—å–Ω–æ–º –ø—Ä–∏–ª–æ–∂–µ–Ω–∏–∏ –∑–¥–µ—Å—å –±—ã–ª–æ –±—ã –æ–±–Ω–æ–≤–ª–µ–Ω–∏–µ —Å–æ—Å—Ç–æ—è–Ω–∏—è –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–∞
      // console.log(`Visible range: [${startIndex}, ${endIndex}]`);
      // console.log(`Current scroll top: ${scrollTop}`);
    }
  }

  // –ü–æ–ª—É—á–∏—Ç—å –¥–∞–Ω–Ω—ã–µ –¥–ª—è —Ä–µ–Ω–¥–µ—Ä–∞
  public getRenderedItems(): T[] {
    const { data } = this.props;
    const { visibleStartIndex, visibleEndIndex } = this.state;
    // –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Ç–æ–ª—å–∫–æ —Ç–µ —ç–ª–µ–º–µ–Ω—Ç—ã, –∫–æ—Ç–æ—Ä—ã–µ –Ω–∞—Ö–æ–¥—è—Ç—Å—è –≤ –≤–∏–¥–∏–º–æ–º –¥–∏–∞–ø–∞–∑–æ–Ω–µ
    return data.slice(visibleStartIndex, visibleEndIndex + 1);
  }

  // –ü–æ–ª—É—á–∏—Ç—å —Å—Ç–∏–ª–∏ –¥–ª—è –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä–∞ (–Ω–∞–ø—Ä–∏–º–µ—Ä, –≤—ã—Å–æ—Ç—É –≤—Å–µ–≥–æ —Å–∫—Ä–æ–ª–ª–∞ –∏ —Å–º–µ—â–µ–Ω–∏–µ)
  public getContainerStyles(): { height: number; paddingTop: number; paddingBottom: number } {
    const { itemHeight } = this.props;
    const { visibleStartIndex } = this.state;

    // –û–±—â–∞—è –≤—ã—Å–æ—Ç–∞ –∫–æ–Ω—Ç–µ–Ω—Ç–∞ (–¥–ª—è –ø–æ–ª–æ—Å—ã –ø—Ä–æ–∫—Ä—É—Ç–∫–∏)
    const totalContentHeight = this.totalHeight;
    // –°–º–µ—â–µ–Ω–∏–µ –¥–ª—è –ø–µ—Ä–≤–æ–≥–æ –≤–∏–¥–∏–º–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
    const offsetTop = visibleStartIndex * itemHeight;

    // —á—Ç–æ–±—ã "–∑–∞–ø–æ–ª–Ω–∏—Ç—å" –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–æ –ø–æ—Å–ª–µ –ø–æ—Å–ª–µ–¥–Ω–µ–≥–æ –≤–∏–¥–∏–º–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞.
    return {

      paddingTop: offsetTop, // –°–º–µ—â–µ–Ω–∏–µ –¥–ª—è —Å–æ–∑–¥–∞–Ω–∏—è —ç—Ñ—Ñ–µ–∫—Ç–∞ "–ø—Ä–æ–∫—Ä—É—Ç–∫–∏"

  }
}
```

#### ### –ò–Ω—Ç–µ—Ä—Ñ–µ–π—Å –¥–ª—è —ç–ª–µ–º–µ–Ω—Ç–∞ —Å –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–æ–π –≤—ã—Å–æ—Ç–æ–π

```typescript
// –ò–Ω—Ç–µ—Ä—Ñ–µ–π—Å —ç–ª–µ–º–µ–Ω—Ç–∞ —Å –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç—å—é –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–æ–π –≤—ã—Å–æ—Ç—ã (–µ—Å–ª–∏ –æ–Ω–∞ –∏–∑–≤–µ—Å—Ç–Ω–∞ –∑–∞—Ä–∞–Ω–µ–µ)

// –ü—Ä–æ–ø—Å—ã –¥–ª—è –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–∞ –≤–∏—Ä—Ç—É–∞–ª—å–Ω–æ–≥–æ —Å–ø–∏—Å–∫–∞ —Å –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–æ–π –≤—ã—Å–æ—Ç–æ–π
```

#### ### –†–∞—Å—á–µ—Ç –≤–∏–¥–∏–º–æ–π –æ–±–ª–∞—Å—Ç–∏ —Å –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–æ–π –≤—ã—Å–æ—Ç–æ–π (–∫–æ–Ω—Ü–µ–ø—Ç)

–≠—Ç–æ –∑–Ω–∞—á–∏—Ç–µ–ª—å–Ω–æ —Å–ª–æ–∂–Ω–µ–µ, –∏ –¥–ª—è –ø–æ–ª–Ω–æ–π —Ä–µ–∞–ª–∏–∑–∞—Ü–∏–∏ —Ç—Ä–µ–±—É–µ—Ç—Å—è –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏–µ —Å DOM (–Ω–∞–ø—Ä–∏–º–µ—Ä, `ResizeObserver` –¥–ª—è –∏–∑–º–µ—Ä–µ–Ω–∏—è –≤—ã—Å–æ—Ç—ã), —á—Ç–æ –≤—ã—Ö–æ–¥–∏—Ç –∑–∞ —Ä–∞–º–∫–∏ —á–∏—Å—Ç–æ TypeScript-—É—Ä–æ–∫–∞. –û–¥–Ω–∞–∫–æ –º—ã –º–æ–∂–µ–º —Å–º–æ–¥–µ–ª–∏—Ä–æ–≤–∞—Ç—å –ª–æ–≥–∏–∫—É.

```typescript
// –ú—ã –±—É–¥–µ–º —Ö—Ä–∞–Ω–∏—Ç—å –∫—ç—à–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ –≤—ã—Å–æ—Ç—ã –∏ –∏—Ö –ø—Ä–µ—Ñ–∏–∫—Å–Ω—ã–µ —Å—É–º–º—ã
interface ItemMetrics {
  height: number;
  offset: number; // –°–º–µ—â–µ–Ω–∏–µ —ç—Ç–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞ –æ—Ç –Ω–∞—á–∞–ª–∞ —Å–ø–∏—Å–∫–∞
}

// –ö–ª–∞—Å—Å –¥–ª—è —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è –º–µ—Ç—Ä–∏–∫–∞–º–∏ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ —Å –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–æ–π –≤—ã—Å–æ—Ç–æ–π
class DynamicHeightListManager<T extends DynamicListItem> {
  private props: DynamicVirtualListProps<T>;
  private itemMetricsMap: Map<string, ItemMetrics> = new Map(); // –ö—ç—à –º–µ—Ç—Ä–∏–∫ –ø–æ ID
  private totalHeight: number = 0; // –ê–∫—Ç—É–∞–ª—å–Ω–∞—è –æ–±—â–∞—è –≤—ã—Å–æ—Ç–∞
  private estimatedItemHeight: number;

  constructor(props: DynamicVirtualListProps<T>) {
    this.props = props;
    this.estimatedItemHeight = props.estimatedItemHeight || 50; // –°—Ä–µ–¥–Ω—è—è/–æ—Ü–µ–Ω–æ—á–Ω–∞—è –≤—ã—Å–æ—Ç–∞
    this.recalculateMetrics(props.data);
  }

  // –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –ø—Ä–æ–ø—Å–æ–≤ –∏ –ø–µ—Ä–µ—Å—á–µ—Ç –º–µ—Ç—Ä–∏–∫
  public updateProps(newProps: DynamicVirtualListProps<T>): void {
    this.props = newProps;
    this.estimatedItemHeight = newProps.estimatedItemHeight || this.estimatedItemHeight;
    this.recalculateMetrics(newProps.data);
  }

  // –ú–µ—Ç–æ–¥ –¥–ª—è –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è –º–µ—Ç—Ä–∏–∫ –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞ (–≤—ã–∑—ã–≤–∞–µ—Ç—Å—è –ø–æ—Å–ª–µ –µ–≥–æ —Ä–µ–Ω–¥–µ—Ä–∞ –∏ –∏–∑–º–µ—Ä–µ–Ω–∏—è)
  public setItemHeight(itemId: string, height: number): void {
    const currentMetric = this.itemMetricsMap.get(itemId);
    if (!currentMetric || currentMetric.height !== height) {
      // –û–±–Ω–æ–≤–ª—è–µ–º –≤—ã—Å–æ—Ç—É –∏ –ø–µ—Ä–µ—Å—á–∏—Ç—ã–≤–∞–µ–º –≤—Å–µ —Å–º–µ—â–µ–Ω–∏—è –ø–æ—Å–ª–µ —ç—Ç–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
      this.itemMetricsMap.set(itemId, { ...currentMetric, height });
      this.recalculateOffsetsFrom(itemId);
    }
  }

  // –ü–µ—Ä–µ—Å—á–∏—Ç—ã–≤–∞–µ–º –≤—Å–µ –º–µ—Ç—Ä–∏–∫–∏ –∏–ª–∏ —Å –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–Ω–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
  private recalculateMetrics(data: T[], startIndexId?: string): void {
    let currentOffset = 0;
    let foundStartIndex = false;

    // –ï—Å–ª–∏ —É–∫–∞–∑–∞–Ω startIndexId, –∏—â–µ–º –µ–≥–æ
    if (startIndexId) {
      const metric = this.itemMetricsMap.get(startIndexId);
      if (metric) {
        currentOffset = metric.offset; // –ù–∞—á–∏–Ω–∞–µ–º –æ—Ç —Å–º–µ—â–µ–Ω–∏—è —ç—Ç–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
        foundStartIndex = true;
      }
    }

    data.forEach((item, index) => {
      let itemHeight = this.estimatedItemHeight;
      const cachedMetric = this.itemMetricsMap.get(item.id);

      // –ï—Å–ª–∏ —ç–ª–µ–º–µ–Ω—Ç —É–∂–µ –±—ã–ª –∏–∑–º–µ—Ä–µ–Ω, –∏—Å–ø–æ–ª—å–∑—É–µ–º –µ–≥–æ —Ä–µ–∞–ª—å–Ω—É—é –≤—ã—Å–æ—Ç—É
      if (cachedMetric && (!startIndexId || foundStartIndex || item.id === startIndexId)) {
        itemHeight = cachedMetric.height;
      } else if (item.height) { // –ï—Å–ª–∏ –≤—ã—Å–æ—Ç–∞ –∑–∞–¥–∞–Ω–∞ –≤ –¥–∞–Ω–Ω—ã—Ö
        itemHeight = item.height;
      }

      this.itemMetricsMap.set(item.id, {
        height: itemHeight,
        offset: currentOffset,
      });
      currentOffset += itemHeight;
    });
    this.totalHeight = currentOffset;
  }

  private recalculateOffsetsFrom(itemId: string): void {
    const data = this.props.data;
    let currentOffset = 0;
    let foundStartItem = false;

    for (const item of data) {
      const cachedMetric = this.itemMetricsMap.get(item.id);
      if (cachedMetric) {
        if (!foundStartItem && item.id !== itemId) {
          // –ò—â–µ–º —ç–ª–µ–º–µ–Ω—Ç, —Å –∫–æ—Ç–æ—Ä–æ–≥–æ –Ω—É–∂–Ω–æ –ø–µ—Ä–µ—Å—á–∏—Ç—ã–≤–∞—Ç—å
          currentOffset = cachedMetric.offset + cachedMetric.height;
          continue;
        } else if (item.id === itemId) {
          foundStartItem = true;
          // –û–±–Ω–æ–≤–ª—è–µ–º offset –¥–ª—è —Ç–µ–∫—É—â–µ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞, –µ—Å–ª–∏ –æ–Ω –∏–∑–º–µ–Ω–∏–ª—Å—è
          this.itemMetricsMap.set(item.id, { ...cachedMetric, offset: currentOffset });
          currentOffset += cachedMetric.height;
          continue;
        }
        // –î–ª—è –≤—Å–µ—Ö –ø–æ—Å–ª–µ–¥—É—é—â–∏—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤, –Ω–∞—á–∏–Ω–∞—è —Å itemId, –ø–µ—Ä–µ—Å—á–∏—Ç—ã–≤–∞–µ–º offset
        this.itemMetricsMap.set(item.id, { ...cachedMetric, offset: currentOffset });
        currentOffset += cachedMetric.height;
      } else {
        // –ï—Å–ª–∏ –º–µ—Ç—Ä–∏–∫ –Ω–µ—Ç, –∏—Å–ø–æ–ª—å–∑—É–µ–º –æ—Ü–µ–Ω–æ—á–Ω—É—é –≤—ã—Å–æ—Ç—É
        const itemHeight = item.height || this.estimatedItemHeight;
        this.itemMetricsMap.set(item.id, { height: itemHeight, offset: currentOffset });
        currentOffset += itemHeight;
      }
    }
    this.totalHeight = currentOffset;
  }


  // –ù–∞–π—Ç–∏ –∏–Ω–¥–µ–∫—Å —ç–ª–µ–º–µ–Ω—Ç–∞, –∫–æ—Ç–æ—Ä—ã–π –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å –≤–∏–¥–µ–Ω –Ω–∞ scrollTop
  public getStartIndexByScrollTop(scrollTop: number): number {
    // –ò—Å–ø–æ–ª—å–∑—É–µ–º –±–∏–Ω–∞—Ä–Ω—ã–π –ø–æ–∏—Å–∫ –¥–ª—è —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ–≥–æ –Ω–∞—Ö–æ–∂–¥–µ–Ω–∏—è startIndex
    let low = 0;
    let high = this.props.data.length - 1;
    let resultIndex = 0;

    while (low <= high) {
      const mid = Math.floor((low + high) / 2);
      const item = this.props.data[mid];
      const metric = this.itemMetricsMap.get(item.id);
      if (metric && metric.offset <= scrollTop) {
        resultIndex = mid;
        low = mid + 1;
      } else {
        high = mid - 1;
      }
    }
    return resultIndex;
  }

  // –û–ø—Ä–µ–¥–µ–ª–∏—Ç—å –≤–∏–¥–∏–º—ã–π –¥–∏–∞–ø–∞–∑–æ–Ω –¥–ª—è –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–æ–π –≤—ã—Å–æ—Ç—ã
  public getVisibleRange(scrollTop: number, containerHeight: number, overscanCount: number = 3): { startIndex: number; endIndex: number } {
    const data = this.props.data;
    if (data.length === 0) return { startIndex: 0, endIndex: 0 };

    let startIndex = this.getStartIndexByScrollTop(scrollTop);
    let currentHeight = 0;
    let endIndex = startIndex;

    // –ò–¥–µ–º –≤–ø–µ—Ä–µ–¥, –ø–æ–∫–∞ –Ω–µ –∑–∞–ø–æ–ª–Ω–∏–º –≤–∏–¥–∏–º—É—é –æ–±–ª–∞—Å—Ç—å
    while (endIndex < data.length) {
      const item = data[endIndex];
      const metric = this.itemMetricsMap.get(item.id);
      currentHeight += metric ? metric.height : this.estimatedItemHeight;
      if (currentHeight >= containerHeight) {
        break;
      }
      endIndex++;
    }

    // –ü—Ä–∏–º–µ–Ω—è–µ–º overscan
    startIndex = Math.max(0, startIndex - overscanCount);
    endIndex = Math.min(data.length - 1, endIndex + overscanCount);

    return { startIndex, endIndex };
  }

  public getTotalHeight(): number {
    return this.totalHeight;
  }

  public getItemOffset(index: number): number {
    const item = this.props.data[index];
    const metric = this.itemMetricsMap.get(item.id);
    return metric ? metric.offset : 0; // –ï—Å–ª–∏ –º–µ—Ç—Ä–∏–∫–∏ –Ω–µ—Ç, —Ç–æ —ç—Ç–æ –æ—à–∏–±–∫–∞ –∏–ª–∏ —ç–ª–µ–º–µ–Ω—Ç –µ—â–µ –Ω–µ –∏–∑–º–µ—Ä–µ–Ω
  }
}
```
