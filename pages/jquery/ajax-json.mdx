<Playground
  template="static"
  files={{
    "/index.html": `<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
  <style>
    body { background: #1e1e2e; color: #cdd6f4; font-family: sans-serif; padding: 20px; margin: 0; }
    button { background: #89b4fa; color: #1e1e2e; border: none; padding: 8px 16px; border-radius: 6px; cursor: pointer; margin: 4px; font-size: 14px; }
    button:hover { background: #74c7ec; }
    .box { background: #313244; border-radius: 8px; padding: 16px; margin: 10px 0; }
    .card { background: #45475a; border-radius: 6px; padding: 10px 14px; margin: 6px 0; }
    .card strong { color: #89b4fa; }
    .tag { background: #313244; border-radius: 4px; padding: 2px 7px; font-size: 12px; margin: 2px; display: inline-block; color: #cba6f7; }
  </style>
</head>
<body>
  <h3 style="color:#cba6f7">Работа с JSON данными</h3>
  <div class="box">
    <button id="loadUsers">Загрузить пользователей</button>
    <button id="loadProducts">Загрузить товары</button>
    <button id="clearData">Очистить</button>
  </div>
  <div id="container" class="box" style="min-height:80px; color:#585b70">Данные появятся здесь...</div>
  <script>
    var mockDB = {
      users: [
        { id: 1, name: 'Алиса', role: 'admin', active: true },
        { id: 2, name: 'Борис', role: 'editor', active: false },
        { id: 3, name: 'Вера',  role: 'viewer', active: true }
      ],
      products: [
        { id: 1, title: 'Ноутбук', price: 75000, tags: ['tech', 'sale'] },
        { id: 2, title: 'Мышь',    price: 1500,  tags: ['tech', 'new'] },
        { id: 3, title: 'Стол',    price: 12000, tags: ['office'] }
      ]
    };

    function mockGetJSON(key, cb) {
      var $c = $('#container').html('<em style="color:#f9e2af">⏳ $.getJSON() загружает...</em>');
      setTimeout(function() { cb(mockDB[key]); }, 700);
    }

    $('#loadUsers').click(function() {
      mockGetJSON('users', function(data) {
        var html = '<strong style="color:#a6e3a1">Пользователи (JSON):</strong>';
        $.each(data, function(i, u) {
          html += '<div class="card"><strong>#' + u.id + ' ' + u.name + '</strong> — ' + u.role +
            ' <span style="color:' + (u.active ? '#a6e3a1' : '#f38ba8') + '">' +
            (u.active ? '● активен' : '● неактивен') + '</span></div>';
        });
        $('#container').html(html);
      });
    });

    $('#loadProducts').click(function() {
      mockGetJSON('products', function(data) {
        var html = '<strong style="color:#a6e3a1">Товары (JSON):</strong>';
        $.each(data, function(i, p) {
          var tags = $.map(p.tags, function(t) { return '<span class="tag">' + t + '</span>'; }).join('');
          html += '<div class="card"><strong>' + p.title + '</strong> — ' + p.price.toLocaleString() + ' ₽ ' + tags + '</div>';
        });
        $('#container').html(html);
      });
    });

    $('#clearData').click(function() {
      $('#container').html('<span style="color:#585b70">Данные очищены</span>');
    });
  </script>
</body>
</html>`,
  }}
/>
