# Цепочки методов

jQuery позволяет вызывать несколько методов подряд.

## Принцип работы

Как работает chaining

```javascript
// Пример кода для Принцип работы
$('.example').text('Hello from Принцип работы');
```

## Примеры цепочек

Практические примеры

```javascript
// Пример кода для Примеры цепочек
$('.example').text('Hello from Примеры цепочек');
```

## .end()

Возврат к предыдущей выборке

```javascript
// Пример кода для .end()
$('.example').text('Hello from .end()');
```

## .addBack()

Добавление предыдущей выборки

```javascript
// Пример кода для .addBack()
$('.example').text('Hello from .addBack()');
```

## Читаемость

Как писать понятные цепочки

```javascript
// Пример кода для Читаемость
$('.example').text('Hello from Читаемость');
```

  files={{
    "/index.html": `<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
  <style>
    body { background: #1e1e2e; color: #cdd6f4; font-family: sans-serif; padding: 20px; margin: 0; }
    button { background: #89b4fa; color: #1e1e2e; border: none; padding: 7px 13px; border-radius: 6px; cursor: pointer; margin: 3px; font-size: 13px; }
    button:hover { background: #74c7ec; }
    .box { background: #313244; border-radius: 8px; padding: 16px; margin: 10px 0; }
    #box { background: #45475a; border-radius: 8px; padding: 20px 24px; display: inline-block; transition: all 0.3s; font-size: 15px; }
    #box p { margin: 6px 0; padding: 6px 10px; border-radius: 4px; background: #313244; }
    .highlight { background: #89b4fa !important; color: #1e1e2e !important; }
    .output { background: #181825; border-radius: 6px; padding: 10px; font-family: monospace; font-size: 12px; color: #cba6f7; min-height: 30px; margin-top: 8px; }
  </style>
</head>
<body>
  <h3 style="color:#cba6f7">Цепочки методов (Chaining)</h3>
  <div class="box">
    <div id="box">
      <p id="p1">Абзац 1</p>
      <p id="p2">Абзац 2</p>
      <p id="p3">Абзац 3</p>
    </div>
  </div>
  <div class="box">
    <button id="chain1">Цепочка 1: css + text + addClass</button>
    <button id="chain2">Цепочка 2: fadeOut + fadeIn</button>
    <button id="chain3">Цепочка 3: find + filter + css</button>
    <button id="chain4">Цепочка 4: .end()</button>
    <button id="reset">Сбросить</button>
    <div class="output" id="chainLog">Нажмите кнопку, чтобы увидеть цепочку...</div>
  </div>
  <script>
    function log(code) { $('#chainLog').text(code); }

    $('#chain1').click(function() {
      $('#box')
        .css('border', '2px solid #89b4fa')
        .find('p')
        .css('color', '#f9e2af')
        .text('Изменён цепочкой!')
        .addClass('highlight');
      log("$('#box').css(...).find('p').css(...).text(...).addClass(...)");
    });

    $('#chain2').click(function() {
      $('#box').fadeOut(400).fadeIn(600);
      log("$('#box').fadeOut(400).fadeIn(600)");
    });

    $('#chain3').click(function() {
      $('#box')
        .find('p')
        .filter(':odd')
        .css({ background: '#313244', color: '#cba6f7' });
      log("$('#box').find('p').filter(':odd').css({...})");
    });

    $('#chain4').click(function() {
      $('#box')
        .find('p')
        .css('color', '#a6e3a1')
        .end()
        .css('border', '2px solid #a6e3a1');
      log("$('#box').find('p').css(...).end().css(...) — .end() возвращает к #box");
    });

    $('#reset').click(function() {
      $('#box').stop(true).show().css({ border: 'none', opacity: 1 });
      $('#box p').removeClass('highlight').css({ color: '', background: '', fontWeight: '' }).each(function(i) {
        $(this).text('Абзац ' + (i + 1));
      });
      log('Сброшено');
    });
  </script>
</body>
</html>`,
  }}
/>
