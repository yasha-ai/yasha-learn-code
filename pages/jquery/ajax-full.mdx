    $('.btn-send').click(function() {
      const method = $('#method').val();
      const url = $('#url').val();
      const dataType = $('#dataType').val();
      const dataText = $('#data').val().trim();
      const headerKey = $('#headerKey').val().trim();
      const headerValue = $('#headerValue').val().trim();
      
      if (!url) {
        alert('–í–≤–µ–¥–∏—Ç–µ URL!');
        return;
      }
      
      // –ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –¥–∞–Ω–Ω—ã—Ö
      let requestData = null;
      if (dataText && (method === 'POST' || method === 'PUT')) {
        try {
          requestData = JSON.parse(dataText);
        } catch (e) {
          alert('–û—à–∏–±–∫–∞ –≤ JSON –¥–∞–Ω–Ω—ã—Ö!');
          return;
        }
      }
      
      // –ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞ headers
      const headers = {};
      if (headerKey && headerValue) {
        headers[headerKey] = headerValue;
      }
      
      // –û—á–∏—Å—Ç–∫–∞ –ª–æ–≥–∞
      $('#log').html('');
      $('#output').addClass('show');
      
      logMessage('info', \`–û—Ç–ø—Ä–∞–≤–∫–∞ \${method} –∑–∞–ø—Ä–æ—Å–∞ –Ω–∞ \${url}\`);
      
      // –ü–æ–ª–Ω–∞—è –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏—è $.ajax()
      $.ajax({
        url: url,
        method: method,
        data: requestData,
        dataType: dataType,
        contentType: 'application/json',
        headers: headers,
        timeout: 10000,
        
        // beforeSend callback
        beforeSend: function(jqXHR, settings) {
          logMessage('info', 'beforeSend: –ó–∞–ø—Ä–æ—Å –≥–æ—Ç–æ–≤ –∫ –æ—Ç–ø—Ä–∞–≤–∫–µ');
          if (Object.keys(headers).length > 0) {
            logMessage('info', 'Custom headers: ' + JSON.stringify(headers));
          }
        }
      })
      // Promise-style handlers
      .done(function(data, textStatus, jqXHR) {
        logMessage('success', \`‚úì success callback: –°—Ç–∞—Ç—É—Å \${jqXHR.status} \${jqXHR.statusText}\`);
        logMessage('success', \`textStatus: \${textStatus}\`);
        
        const preview = typeof data === 'string' ? data.substring(0, 200) : JSON.stringify(data, null, 2).substring(0, 500);
        $('#log').append(\`
          <div class="response-data">
            <strong>Response Data:</strong><br>
            <pre>\${preview}\${preview.length >= 200 ? '...' : ''}</pre>
          </div>
        \`);
      })
      .fail(function(jqXHR, textStatus, errorThrown) {
        logMessage('error', \`‚úó fail callback: \${textStatus}\`);
        logMessage('error', \`Error: \${errorThrown || 'Unknown error'}\`);
        if (jqXHR.status) {
          logMessage('error', \`HTTP Status: \${jqXHR.status}\`);
        }
      })
      .always(function(dataOrJqXHR, textStatus, jqXHROrErrorThrown) {
        logMessage('info', 'always callback: –ó–∞–ø—Ä–æ—Å –∑–∞–≤–µ—Ä—à–µ–Ω (—É—Å–ø–µ—Ö –∏–ª–∏ –æ—à–∏–±–∫–∞)');
