// Plugin 1: Highlight
(function($) {
  $.fn.highlight = function(options) {
    const settings = $.extend({
      color: '#fbbf24',
      duration: 500
    }, options);
    
    return this.each(function() {
      const $el = $(this);
      const originalBg = $el.css('background-color');
      
      $el.stop(true).css('background-color', settings.color)
        .animate({ backgroundColor: originalBg }, settings.duration * 2);
    });
  };
})(jQuery);

// Plugin 2: Tooltip
(function($) {
  $.fn.tooltip = function(options) {
    const settings = $.extend({
      showDelay: 200
    }, options);
    
    return this.each(function() {
      const $el = $(this);
      const text = $el.data('tooltip');

})(jQuery);
