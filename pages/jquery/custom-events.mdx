# Кастомные события

Создание собственных событий для модульной архитектуры.

## .trigger()

Инициирование событий

```javascript
// Пример кода для .trigger()
$('.example').text('Hello from .trigger()');
```

## Кастомные данные

Передача параметров

```javascript
// Пример кода для Кастомные данные
$('.example').text('Hello from Кастомные данные');
```

## Namespace events

Именование событий

```javascript
// Пример кода для Namespace events
$('.example').text('Hello from Namespace events');
```

## Интерактивный пример

<Playground
  template="static"
  files={{
    "/index.html": `<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
  <style>
    body { background: #1e1e2e; color: #cdd6f4; font-family: sans-serif; padding: 20px; margin: 0; }
    button { background: #89b4fa; color: #1e1e2e; border: none; padding: 7px 13px; border-radius: 6px; cursor: pointer; margin: 3px; font-size: 13px; }
    button:hover { background: #74c7ec; }
    button.off { background: #f38ba8; }
    .box { background: #313244; border-radius: 8px; padding: 16px; margin: 10px 0; }
    .log-item { background: #45475a; border-radius: 4px; padding: 6px 10px; margin: 3px 0; font-family: monospace; font-size: 12px; border-left: 3px solid #89b4fa; }
    .log-item.ns { border-left-color: #cba6f7; }
    .log-item.data { border-left-color: #a6e3a1; }
    #cart-count { background: #f38ba8; color: #1e1e2e; border-radius: 50%; padding: 2px 7px; font-size: 12px; font-weight: bold; }
    #notify { background: #45475a; padding: 8px 14px; border-radius: 6px; display: none; margin: 6px 0; }
  </style>
</head>
<body>
  <h3 style="color:#cba6f7">Кастомные события jQuery</h3>

  <div class="box">
    <strong>Корзина: </strong><span id="cart-count">0</span> товаров
    <div id="notify"></div>
    <br>
    <button id="addToCart">Добавить товар → .trigger('cart:add')</button>
    <button id="clearCart">Очистить → .trigger('cart:clear')</button>
    <button id="triggerNs">.trigger('ui.refresh')</button>
    <button id="triggerData">.trigger('user:login', {name:'Алиса'})</button>
    <button id="offEvent" class="off">Отписать cart:add</button>
  </div>

  <div class="box">
    <strong style="font-size:12px;color:#bac2de">Лог событий:</strong>
    <div id="eventLog"></div>
  </div>

  <script>
    var cartCount = 0;
    var cartAddOn = true;

    function logEvent(msg, type) {
      var $item = $('<div class="log-item ' + (type || '') + '">' + new Date().toLocaleTimeString() + ' — ' + msg + '</div>');
      $('#eventLog').prepend($item);
      if ($('#eventLog .log-item').length > 6) $('#eventLog .log-item:last').remove();
    }

    // Подписка на кастомное событие
    $(document).on('cart:add', function(e, item) {
      if (!cartAddOn) return;
      cartCount++;
      $('#cart-count').text(cartCount);
      $('#notify').text('✅ Добавлено: ' + (item ? item.name : 'Товар')).fadeIn(200).delay(1500).fadeOut(400);
      logEvent('cart:add — корзина: ' + cartCount + ' шт', '');
    });

    $(document).on('cart:clear', function() {
      cartCount = 0;
      $('#cart-count').text(0);
      logEvent('cart:clear — корзина очищена', '');
    });

    // Namespace событие
    $(document).on('ui.refresh', function() {
      logEvent('ui.refresh — интерфейс обновлён', 'ns');
    });

    // Событие с данными
    $(document).on('user:login', function(e, data) {
      logEvent('user:login — привет, ' + data.name + '!', 'data');
    });

    $('#addToCart').click(function() {
      $(document).trigger('cart:add', [{ name: 'Товар #' + (cartCount + 1) }]);
    });

    $('#clearCart').click(function() {
      $(document).trigger('cart:clear');
    });

    $('#triggerNs').click(function() {
      $(document).trigger('ui.refresh');
    });

    $('#triggerData').click(function() {
      $(document).trigger('user:login', [{ name: 'Алиса' }]);
    });

    $('#offEvent').click(function() {
      if (cartAddOn) {
        $(document).off('cart:add');
        cartAddOn = false;
        $(this).text('Подписать cart:add').removeClass('off').css('background','#a6e3a1');
        logEvent('.off("cart:add") — обработчик удалён', '');
      } else {
        $(document).on('cart:add', function(e, item) {
          cartCount++;
          $('#cart-count').text(cartCount);
          logEvent('cart:add (re-on) — корзина: ' + cartCount, '');
        });
        cartAddOn = true;
        $(this).text('Отписать cart:add').addClass('off').css('background','');
        logEvent('.on("cart:add") — обработчик добавлен снова', '');
      }
    });
  </script>
</body>
</html>`,
  }}
/>
